non_colonial_power_controlled_african_states = {
	input = collection:african_states 
	operators = {
		limit = {
			controller = {
				AND = {
					capital_scope = {
						is_on_continent = africa 
					}
					NOT = {
						any_country = {
							has_subject = PREV
							capital_scope = {
								NOT = {
									is_on_continent = africa
								}
							}
						}
					}
				}
			}
		}
	}
	name = COLLECTION_NON_COLONIAL_CONTROLLED_AFRICAN_STATES
}
