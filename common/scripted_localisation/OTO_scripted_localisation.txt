defined_text = {
	name = GetVidinSeparatism
	text = {
		trigger = {
			original_tag = OTO
			NOT = {
				has_country_flag = OTO_vidin_being_hellenized_flag
			}
		}
		localization_key = OTO_vidin_separatism_1
	}
	text = {
		trigger = {
			AND = {
				has_country_flag = OTO_vidin_being_hellenized_flag
			}
		}
		localization_key = OTO_vidin_separatism_2
	}
	text = {
		trigger = {
			check_variable = {
				var = OTO_vidin_separatism
				value = 100
				compare = greater_than_or_equals
			}
		}
		localization_key = OTO_vidin_locked
	}
}
