	afr_annex_senegal={
		available = { }
		visible={
			has_country_flag = BC_Granted_Senegal
			NOT={has_global_flag = BC_Senegal_Colonised}
		}
		cost = 50
		fire_only_once = yes
		days_remove = 240
		remove_effect = {
			set_global_flag = BC_Senegal_Colonised
			every_state={#tribes are subsumed
				limit={
					any_state_of = { target = constant:state_groups.senegal }
					owner={
						OR={
							tag = AFR
							tag = UCT 
						}
					}
					owner={ is_ai = yes }
				}
				if={
					limit={
						any_country={
							is_subject_of = ROOT
							capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
						}
					}
					random_country={
						limit={
							is_subject_of = ROOT
							capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
						}
						transfer_state = PREV
						PREV={add_compliance = 50}
					}
				}
				else={
					ROOT={transfer_state = PREV}
					add_compliance = 50
				}
			}
			every_state={#the border states of resistors are annexed, but with high resistance
				limit={
					any_state_of = { target = constant:state_groups.senegal }
					
					is_capital = no
					owner={ OR={ is_ai = yes } }
				}
				if={
					limit={
						any_country={
							is_subject_of = ROOT
							capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
						}
					}
					random_country={
						limit={
							is_subject_of = ROOT
							capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
						}
						transfer_state = PREV
						PREV={add_resistance = 30}
					}
				}
				else={
					ROOT={transfer_state = PREV}
					add_resistance = 30
				}
			}
			every_state={#declare war on resistor capitals, due to cores if they win they should get thier land back, will likely script the peace later
				limit={
					is_capital = yes
					owner={ OR={ is_ai = yes } }
					any_state_of = { target = constant:state_groups.senegal }
					
				}
				if={
					limit={
						any_country={
							is_subject_of = ROOT
							capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
						}
					}
					owner={
						random_country={
							limit={
								is_subject_of = ROOT
								capital_scope={ any_state_of = { target = constant:state_groups.senegal } }
							}
							declare_war_on = {
								target = PREV
								type = take_claimed_state
							}
							ROOT={
								add_to_war = {
									targeted_alliance = PREV 
									enemy = PREV.PREV 
									hostility_reason = asked_to_join
								}
							}
						}
					}
				}
				else={
					owner={
						ROOT={
							declare_war_on = {
								target = PREV
								type = take_claimed_state
							}
						}
					}
				}
			}
		}
		ai_will_do = {  factor = 16 }
	}
