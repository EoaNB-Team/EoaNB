	afr_annex_senegal = {
		visible = {
			has_country_flag = BC_Granted_Senegal
			NOT = { has_global_flag = BC_Senegal_Colonised }
		}
		cost = 50
		fire_only_once = yes
		days_remove = 240
		remove_effect = {
			set_global_flag = BC_Senegal_Colonised
			set_temp_variable = { AFR_has_regional_subject = 0 }
			if = {
				limit = {
					any_country = {
						is_subject_of = ROOT
						capital_scope = { any_state_of = { target = constant:state_groups.senegal } }
					}
				}
				random_country = {
					limit = {
						is_subject_of = ROOT
						capital_scope = { any_state_of = { target = constant:state_groups.senegal } }
					}
					save_event_target_as = AFR_regional_subject
				}
				set_temp_variable = { AFR_has_regional_subject = 1 }
			}
			every_state = {
				limit = {
					any_state_of = { target = constant:state_groups.senegal }
					owner = {
						OR = { tag = AFR tag = UCT }
						is_ai = yes
					}
				}
				AFR_colonise_subsume_state = yes
			}
			every_state = {
				limit = {
					any_state_of = { target = constant:state_groups.senegal }
					is_capital = no
					owner = { is_ai = yes }
				}
				AFR_colonise_annex_border_state = yes
			}
			every_state = {
				limit = {
					any_state_of = { target = constant:state_groups.senegal }
					is_capital = yes
					owner = { is_ai = yes }
				}
				AFR_colonise_declare_war_on_capital_owner = yes
			}
		}
		ai_will_do = { factor = 16 }
	}
