	else_if = {
		limit = { check_variable = { v = token:partial_economic_mobilisation } }
		OR = {
			has_idea = low_economic_mobilisation
			has_idea = war_economy
		}
	}
	else_if = {
		limit = { check_variable = { v = token:war_economy } }
		OR = {
			has_idea = partial_economic_mobilisation
			has_idea = tot_economic_mobilisation
		}
	}
	else_if = {
		limit = { check_variable = { v = token:tot_economic_mobilisation } }
		has_idea = war_economy
	}
	else_if = {
		limit = { check_variable = { v = token:cl_legal } }
		has_idea = cl_restricted
	}
	else_if = {
		limit = { check_variable = { v = token:cl_restricted } }
		OR = {
			has_idea = cl_legal
			has_idea = cl_illegal
		}
	}
	else_if = {
		limit = { check_variable = { v = token:cl_illegal } }
		has_idea = cl_restricted
	}
}
