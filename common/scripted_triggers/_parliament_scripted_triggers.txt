		has_idea = low_level_mobilisation
	}
	else_if = {
		limit = { check_variable = { v = token:low_level_mobilisation } }
		OR = {
			has_idea = demobilised_economy
			has_idea = partial_mobilisation
		}
	}
	else_if = {
		limit = { check_variable = { v = token:partial_mobilisation } }
		OR = {
			has_idea = low_level_mobilisation
			has_idea = war_economy
		}
	}
	else_if = {
		limit = { check_variable = { v = token:war_economy } }
		OR = {
			has_idea = partial_mobilisation
			has_idea = total_mobilisation
		}
	}
	else_if = {
		limit = { check_variable = { v = token:total_mobilisation } }
		has_idea = war_economy
	}
	else_if = {
		limit = { check_variable = { v = token:legal_child_labour } }
		has_idea = restricted_child_labour
	}
	else_if = {
		limit = { check_variable = { v = token:restricted_child_labour } }
		OR = {
			has_idea = legal_child_labour
			has_idea = banned_child_labour
		}
	}
	else_if = {
		limit = { check_variable = { v = token:banned_child_labour } }
		has_idea = restricted_child_labour
	}
}
