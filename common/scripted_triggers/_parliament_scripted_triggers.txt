# Sees if you can do the reform (If the law is before or after the current one):
parliament_reform_check = {
	if = {
		limit = { check_variable = { v = token:no_elections } }
		OR = {
			has_government = autocracy
			has_government = fundamentalism
			has_government = ideology_radical_socialism
			has_government = chauvinist_populism
		}
		NOT = { has_idea = no_elections }
	}
	else_if = {
		limit = { check_variable = { v = token:landed_voting } }
		OR = {
			has_idea = no_elections
			has_idea = weighted_wealth_voting
		}
	}
	else_if = {
		limit = { check_variable = { v = token:weighted_wealth_voting } }
		OR = {
			has_idea = no_elections
			has_idea = landed_voting
		}
	}
	else_if = {
		limit = { check_variable = { v = token:universal_male_suffrage } }
		NOT = {
			OR = {
				has_government = autocracy
				has_government = fundamentalism
				has_government = ideology_radical_socialism
				has_government = chauvinist_populism
			}
		}
		OR = {
			has_idea = l
			has_idea = universal_male_suffrage
		}
	}
	else_if = {
		limit = { check_variable = { v = token:universal_suffrage } }
		NOT = {
			OR = {
				has_government = autocracy
				has_government = fundamentalism
				has_government = ideology_radical_socialism
