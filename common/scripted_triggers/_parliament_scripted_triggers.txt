# Sees if you can do the reform (If the law is before or after the current one):
parliament_reform_check = {
	if = {
		limit = { check_variable = { v = token:no_elections } }
		OR = {
			has_government = autocracy
			has_government = fundamentalism
			has_government = ideology_radical_socialism
			has_government = chauvinist_populism
		}
		NOT = { has_idea = no_elections }
	}
	else_if = {
		limit = { check_variable = { v = token:landed_voting } }
		OR = {
			has_idea = no_elections
			has_idea = weighted_wealth_voting
		}
	}
	else_if = {
		limit = { check_variable = { v = token:weighted_wealth_voting } }
		OR = {
			has_idea = no_elections
			has_idea = landed_voting
			has_idea = wealth_voting
		}
	}
	else_if = {
		limit = { check_variable = { v = token:wealth_voting } }
		OR = {
			has_idea = no_elections
			has_idea = landed_voting
			has_idea = weighted_wealth_voting
		}
	}
	else_if = {
		limit = { check_variable = { v = token:universal_male_suffrage } }
		NOT = {
			OR = {
				has_government = autocracy
				has_government = fundamentalism
				has_government = ideology_radical_socialism
				has_government = chauvinist_populism
			}
		}
		OR = {
			has_idea = wealth_voting
			has_idea = universal_male_suffrage
		}
	}
	else_if = {
		limit = { check_variable = { v = token:universal_suffrage } }
		NOT = {
			OR = {
				has_government = autocracy
				has_government = fundamentalism
				has_government = ideology_radical_socialism
				has_government = chauvinist_populism
			}
		}
		has_idea = womens_suffrage
		has_idea = universal_male_suffrage
	}

	else_if = {
		limit = { check_variable = { v = token:national_supremacy } }
		has_idea = racial_segregation
	}
	else_if = {
		limit = { check_variable = { v = token:racial_segregation } }
		OR = {
			has_idea = national_supremacy
			has_idea = limited_equality
		}
	}
	else_if = {
		limit = { check_variable = { v = token:limited_equality } }
		OR = {
			has_idea = full_equality
			has_idea = racial_segregation
		}
	}
	else_if = {
		limit = { check_variable = { v = token:full_equality } }
		has_idea = limited_equality
	}

	else_if = {
		limit = { check_variable = { v = token:theocracy } }
		has_idea = state_religion
		has_government = fundamentalism
	}
	else_if = {
		limit = { check_variable = { v = token:state_religion } }
		OR = {
			has_idea = freedom_of_religion
			has_idea = theocracy
		}
	}
	else_if = {
		limit = { check_variable = { v = token:freedom_of_religion } }
		OR = {
			has_idea = total_seperation
			has_idea = state_atheism
			has_idea = state_religion
		}
	}
	else_if = {
		limit = { check_variable = { v = token:total_seperation } }
		OR = {
			has_idea = freedom_of_religion
			has_idea = state_atheism
		}
	}
	else_if = {
		limit = { check_variable = { v = token:state_atheism } }
		has_idea = total_seperation
		NOT = { has_government = fundamentalism }
	}

	else_if = {
		limit = { check_variable = { v = token:politicized_courts } }
		OR = {
			has_government = autocracy
			has_government = fundamentalism
			has_government = ideology_radical_socialism
			has_government = chauvinist_populism
		}
		has_idea = corrupt_courthouse
	}
	else_if = {
		limit = { check_variable = { v = token:corrupt_courthouse } }
		OR = {
			has_idea = politicized_courts
			has_idea = aristocratic_courts
		}
	}
	else_if = {
		limit = { check_variable = { v = token:aristocratic_courts } }
		OR = {
			has_idea = politicized_courts
			has_idea = corrupt_courthouse
			has_idea = quasi_jury_court
		}
	}
	else_if = {
		limit = { check_variable = { v = token:quasi_jury_court } }
		OR = {
			has_idea = full_jury_court
			has_idea = aristocratic_courts
		}
	}
	else_if = {
		limit = { check_variable = { v = token:full_jury_court } }
		NOT = {
			OR = {
				has_government = autocracy
				has_government = fundamentalism
				has_government = ideology_radical_socialism
				has_government = chauvinist_populism
			}
		}
		has_idea = quasi_jury_court
	}

	else_if = {
		limit = { check_variable = { v = token:dedicated_police } }
		NOT = { has_idea = dedicated_police }
	}
	else_if = {
		limit = { check_variable = { v = token:police_state } }
		has_idea = dedicated_police
	}

	else_if = {
		limit = { check_variable = { v = token:apothecaries } }
		has_idea = charitable_healthcare
	}
	else_if = {
		limit = { check_variable = { v = token:charitable_healthcare } }
		OR = {
			has_idea = apothecaries
			has_idea = private_healthcare
		}
	}
	else_if = {
		limit = { check_variable = { v = token:private_healthcare } }
		OR = {
			has_idea = charitable_healthcare
			has_idea = public_healthcare
		}
	}
	else_if = {
		limit = { check_variable = { v = token:public_healthcare } }
		has_idea = private_healthcare
	}

	else_if = {
		limit = { check_variable = { v = token:oral_tradition } }
		OR = {
			has_idea = traditional_ulama
			has_idea = church_sponsored_education
		}
	}
	else_if = {
		limit = { check_variable = { v = token:church_sponsored_education } }
		NOT = { is_muslim_country = yes }
		OR = {
			has_idea = private_education
			has_idea = public_schooling_system
			has_idea = oral_tradition
		}
	}
	else_if = {
		limit = { check_variable = { v = token:traditional_ulama } }
		is_muslim_country = yes
		OR = {
			has_idea = private_education
			has_idea = public_schooling_system
			has_idea = oral_tradition
		}
	}
	else_if = {
		limit = { check_variable = { v = token:private_education } }
		OR = {
			has_idea = church_sponsored_education
			has_idea = public_schooling_system
		}
	}
	else_if = {
		limit = { check_variable = { v = token:public_schooling_system } }
		has_idea = private_education
	}

	else_if = {
		limit = { check_variable = { v = token:disarmed_nation } }

			has_idea = disarmed_nation

		limit = { check_variable = { v = token:allotment_system } }

			has_idea = scraping_the_barrel
		}
	}

		limit = { check_variable = { v = token:volunteer_only } }

			has_idea = disarmed_nation
			has_idea = allotment_system
			has_idea = early_conscription
			has_idea = limited_conscription

			has_idea = scraping_the_barrel
		}
	}
	else_if = {

	}
	else_if = {
		limit = { check_variable = { v = token:all_adults_serve } }
		has_war = yes
		OR = {
			has_idea = service_by_requirement
			has_idea = scraping_the_barrel
		}
	}
	else_if = {
