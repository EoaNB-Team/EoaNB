# example usage:
## calculate_inflation_change = yes
calculate_inflation_change = {
	# Flat Inflation Change Modifiers:
	set_temp_variable = { daily_inflation_change = { value = 0  add = modifier@inflation_change } }

	# Clamp Inflation Change Factor Modifiers:
	set_temp_variable = {
		daily_inflation_change_factor = {
			value = 1
			add = modifier@inflation_change_factor
			max = 0											# was: if factor < 0 -> 0
		}
	}
	# Apply Inflation Change Factor Modifiers:
	multiply_temp_variable = { daily_inflation_change = daily_inflation_change_factor }

	# Setting Display Variable:
	set_temp_variable = { daily_inflation_change_factor_display = modifier@inflation_change_factor }
	set_temp_variable = { daily_inflation_change_factor_from_ideas = modifier@inflation_change_factor_from_ideas }
	set_temp_variable = { daily_inflation_change_factor_from_decisions = modifier@inflation_change_factor_from_decisions }
	set_temp_variable = {
		daily_inflation_change_factor_from_other = {
			value = daily_inflation_change_factor_display
			subtract = daily_inflation_change_factor_from_ideas
			subtract = daily_inflation_change_factor_from_decisions
		}
	}
	set_temp_variable = { daily_inflation_change_factor_display = { value = daily_inflation_change_factor_display  max = -1 } }	# was: if display < -1 -> -1
}

calculate_money_income = {
	set_temp_variable = { total_income_temp = 0 }

	# Taxes
	set_temp_variable = { income_from_our_taxes = 0 }
	set_temp_variable = { income_from_our_taxes_core = 0 }
	set_temp_variable = { income_from_our_taxes_non_core = 0 }
	hidden_trigger = {
		all_owned_state = {
			OR = {
				AND = {
					is_core_of = PREV
					add_to_temp_variable = { income_from_our_taxes_core = state_population_k }
				}
				AND = {
					NOT = { is_core_of = PREV }
					add_to_temp_variable = { income_from_our_taxes_non_core = state_population_k }
				}
			}
		}
	}
	divide_temp_variable = { income_from_our_taxes_core = 9000 }
	divide_temp_variable = { income_from_our_taxes_non_core = 300000 }
	set_temp_variable = {
		income_from_our_taxes = {
			value = income_from_our_taxes_core
			add = income_from_our_taxes_non_core
			multiply = 7										# money is now weekly
		}
	}
	if = {													# overflow guard: sets to ~maxint when negative (not a clamp), kept as-is
		limit = { check_variable = { income_from_our_taxes < 0 } }
		set_temp_variable = { income_from_our_taxes = 2147483 }
	}

	# Tax Affecting Modifiers:
	calculate_tax_modifiers = yes

	# Consumer Goods Factories:																										# Example: Prussia
	set_temp_variable = {
		income_from_our_consumer_goods_factories = {
			value = num_of_civilian_factories								# 10 total civs at game start					= income 10
			multiply = modifier@consumer_goods_expected_value				# Expected civs at game start 30% (10 x 0.3)	= income 3
			#round = yes													# Nothing to round here + not necessary			= income 3
			max = 0															# was: if income < 0 -> 0
			multiply = 0.3													# Income decreased to one third (3 x 0.3)		= income 0.9
			multiply = 3													# money is now weekly (0.3 x 3)					= income 2.7
		}
	}

	# Consumer Goods Factories Modifiers:
	calculate_consumer_goods_modifiers = yes																						# There are modifiers that modify the total income from civs, read below.

	# Free Civilian Factories:
	set_temp_variable = {
		income_from_our_free_civilian_factories = {
			value = num_of_available_civilian_factories
			multiply = 0.08
			multiply = 3													# money is now weekly
		}
	}
	add_to_temp_variable = { total_income_temp = income_from_our_free_civilian_factories }

	# Income from tariffs:
	set_temp_variable = {
		income_from_tariffs = {
			value = num_of_factories
			subtract = num_of_owned_factories
		}
	}
	if = {
		limit = { check_variable = { income_from_tariffs > 0 } }
		if = {
			limit = { has_idea = eoanb_tariff_free_trade }
			multiply_temp_variable = { income_from_tariffs = 0.05 }
		}
		else_if = {
			limit = { has_idea = eoanb_tariff_mercantilism }
			multiply_temp_variable = { income_from_tariffs = 0.15 }
		}
		else_if = {
			limit = { has_idea = eoanb_tariff_protectionism }
			multiply_temp_variable = { income_from_tariffs = 0.3 }
		}
		else_if = {
			limit = { has_idea = eoanb_tariff_autarky }
			multiply_temp_variable = { income_from_tariffs = 0.75 }
		}
		else = {
			multiply_temp_variable = { income_from_tariffs = 0.05 } # fallback, no policy
		}
		multiply_temp_variable = { income_from_tariffs = 3 } # weekly basis, matches other lines
	}
	else = {
		set_temp_variable = { income_from_tariffs = 0 }
	}
	add_to_temp_variable = { total_income_temp = income_from_tariffs }

	#multiply_temp_variable = { income_from_trading_minerals = 0.04 }

	# Consumer Goods Factories Modifiers:																							# Example: Prussia

		consumer_goods_income_modifier = {
			value = 1																	# Income modifier at start = 1 (no change)	= income modified by 1

			max = 0																		# was: if modifier < 0 -> 0
		}
	}
	multiply_temp_variable = { income_from_our_consumer_goods_factories = consumer_goods_income_modifier }							# Current income of 2.7 is modified by the factors from ideas etc. - 2.7 x 1.15 = 3.105
	add_to_temp_variable = { total_income_temp = income_from_our_consumer_goods_factories }											# Income of 3.105 is added to the total income

	# Setting Display Variable:
	set_temp_variable = { consumer_goods_modifier = modifier@consumer_goods_income_factor }
	set_temp_variable = { consumer_goods_modifier_from_ideas = modifier@consumer_goods_income_factor_from_ideas }
	set_temp_variable = { consumer_goods_modifier_from_decisions = modifier@consumer_goods_income_factor_from_decisions }
	set_temp_variable = {
		consumer_goods_modifier_from_other = {
			value = consumer_goods_modifier
			subtract = consumer_goods_modifier_from_ideas
			subtract = consumer_goods_modifier_from_advisors		# NOTE: _from_advisors is not set in this effect (external/global?) - verify
			subtract = consumer_goods_modifier_from_decisions
		}
	}
	set_temp_variable = { consumer_goods_modifier = { value = consumer_goods_modifier  max = -1 } }	# was: if modifier < -1 -> -1

## calculate_money_expenses = yes
calculate_money_expenses = {
	set_temp_variable = { total_expenses_temp = 0 }

	# Social Spending:
	set_temp_variable = { expenses_from_our_social_spending = 0 }
	hidden_trigger = {
		all_owned_state = {
			OR = {
				NOT = { is_core_of = PREV }
				AND = {
					is_core_of = PREV
					add_to_temp_variable = { expenses_from_our_social_spending = state_population_k }
				}
			}
		}
	}
	set_temp_variable = {
		expenses_from_our_social_spending = {
			value = expenses_from_our_social_spending
			divide = 2000
			multiply = -0.05
			multiply = 7										# money is now weekly
		}
	}

	# Social Spending Affecting Modifiers:
	calculate_social_spending_modifiers = yes

	# Research and Education Spending:
	set_temp_variable = {
		expenses_from_our_research_spending = {
			value = max_manpower_k								#e.g. Britain has 26.53m population = 26,530k	COST
			divide = 1000										#26,530 / 1000									= 26.53
			divide = 10											#26.53 / 10										= 2.653
			multiply = -0.2										#2.653 x -0.2									= -0.5306
			multiply = amount_research_slots					#-0.5306 x 4									= 2.1224
			multiply = 4										#2.1224 x 4										= 8.4896
		}
	}

	# Research and Education Spending Affecting Modifiers:
	calculate_research_spending_modifiers = yes

	# Army Spending:
	set_temp_variable = {
		expenses_from_our_army_spending = {
			value = num_battalions
			multiply = -0.003
			multiply = 7										# money is now weekly
		}
	}

	# Army Spending Affecting Modifiers:
	calculate_army_spending_modifiers = yes

	# Navy Spending:
	set_temp_variable = {
		expenses_from_our_navy_spending = {
			value = num_ships
			multiply = -0.003
			multiply = 7										# money is now weekly
		}
	}

	# Navy Spending Affecting Modifiers:
	calculate_navy_spending_modifiers = yes

	# Road Spending:
	set_temp_variable = { expenses_from_our_road_spending = 0 }
	hidden_trigger = {
		all_owned_state = {
			OR = {
				NOT = { is_core_of = ROOT }
				AND = {
					is_core_of = ROOT
					add_to_temp_variable = { expenses_from_our_road_spending = infrastructure_level }
				}
			}
		}
	}
	set_temp_variable = {
		expenses_from_our_road_spending = {
			value = expenses_from_our_road_spending
			multiply = -0.002
			multiply = 7										# money is now weekly
		}
	}

	# Road Spending Affecting Modifiers:
	calculate_road_spending_modifiers = yes

	# Law Spending:
	set_temp_variable = {
		expenses_from_our_law_spending = {
			value = max_manpower_k
			divide = 1000
			divide = 10
			multiply = -0.2
			multiply = 7										# money is now weekly
		}
	}

	# Law Spending Affecting Modifiers:
	calculate_law_spending_modifiers = yes

	# Used Military Factories:
	set_temp_variable = {
		expenses_from_our_used_mil_factories = {
			value = num_of_military_factories
			subtract = num_of_available_military_factories
			round = yes
			multiply = -0.085
			multiply = 7										# money is now weekly
		}
	}
	add_to_temp_variable = { total_expenses_temp = expenses_from_our_used_mil_factories }

	# Interests from Loans:
	set_temp_variable = {
		expenses_from_loans = {
			value = current_loans_sum
			divide = -100
		}
	}
	add_to_temp_variable = { total_expenses_temp = expenses_from_loans }

	# Other Expanses:
	set_temp_variable = { expenses_from_other_sources = { value = modifier@money_expenses  min = 0 } }	# was: if expenses > 0 -> 0
	add_to_temp_variable = { total_expenses_temp = expenses_from_other_sources }

	# Expenses Affecting Modifiers:
	calculate_money_expenses_modifiers = yes
}
