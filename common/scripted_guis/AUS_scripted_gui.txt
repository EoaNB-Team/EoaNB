	AUS_challenge_of_multination_empire_decisions_grid_1_gui = {
		context_type = player_context
		window_name = "AUS_challenge_of_multination_empire_decisions_conatiner_grid_1"

		parent_window_name = "AUS_challenge_of_multination_empire_decisions_container"

		visible = {
			has_country_flag = AUS_multinational_empire_decisions_open
			has_country_flag = AUS_state_1_selected
		}

		effects = {
			AUS_multinational_empire_decision_select_click = {
				custom_effect_tooltip = AUS_multinational_empire_enact_decision_tt
				AUS_multinational_empire_enact_decision = yes
			}
		}
		triggers = {
			AUS_multinational_empire_decision_select_click_enabled = {
				NOT = {
					has_country_flag = AUS_ongoing_test
				}
				set_temp_variable = { lpcost = lp_cost@var:mne }
				has_political_power > var:lpcost
				if = {
					limit = { check_variable = { debt_cost@var:mne > 0 } }
					set_temp_variable = { loan_size_to_check = 50}
					has_less_than_specific_loan_size = yes
				}
				check_variable = {
					clicked@var:mne = 0
				}
			}
			AUS_multinational_empire_decision_background_available_visible = {
				set_temp_variable = { lpcost = cost1@var:mne }
				has_political_power > var:lpcost
				check_variable = {
					clicked@var:mne = 0
				}
			}
		}
		properties = {
			AUS_multinational_empire_decision_icon = {
				image = "GetAustrianMNEIcon"
			}
		}
		dynamic_lists = {
			AUS_multinational_empire_decisions_grid = {
				array = AUS_decision_array_1
				entry_container = AUS_multinational_empire_decision_list_entry
			}
		}
	}
