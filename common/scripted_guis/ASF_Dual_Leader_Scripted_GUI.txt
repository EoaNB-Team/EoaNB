scripted_gui = {
	Vice_Leader = {
		window_name = "Vice_Leader"
		context_type = player_context
		parent_window_token = politics_tab
		visible = { 
			is_ai = no
			has_variable = show_VL_window
		}
		triggers = { vice_leader_picture_click_enabled = { set_temp_variable = { temp_portrait = character_token } } }
		effects = {
			fetch_dual_leader_description = {
				meta_effect = {
					text = { custom_effect_tooltip = [CHAR]_desc }
					CHAR = "[?character_token.GetTokenKey]"
				}
			}
			show_VL_button_click = {
                if = {
                    limit = { has_variable = show_VL_window }
                	clear_variable = show_VL_window
                }
                else = { set_variable = { show_VL_window = 1 } }
            }
		}
		properties = { vice_leader_picture = { image = "GFX_portrait_[?temp_portrait.GetTokenKey]" } }
	}
}
