scripted_gui = {
	
	### TOGGLE/SWAP BUTTON ###
	vice_leader_toggle_button_gui = {
		context_type = player_context
		window_name = "vice_leader_toggle_button_container"
		parent_window_token = politics_tab

		visible = {
			is_ai = no
			has_vice_leader = yes
		}
		
		effects = {
			vice_leader_toggle_button_click = {
				if = {
					limit = { has_country_flag = show_vice_leader_window }
					clr_country_flag = show_vice_leader_window
				}
				else = { set_country_flag = show_vice_leader_window }
			}
		}
		
		triggers = { vice_leader_toggle_button_click_enabled = { always = yes } }
	}
	
	### VICE LEADER PORTRAIT WINDOW ###
	vice_leader_window_gui = {
		context_type = player_context
		window_name = "vice_leader_window"
		parent_window_token = politics_tab
		
		visible = {
			is_ai = no
			has_country_flag = show_vice_leader_window
			has_vice_leader = yes
		}
		
		properties = { vice_leader_picture = { image = "[VL_picture]" } }
		
		triggers = { vice_leader_picture_visible = { always = yes } }
	}
	
	### VICE LEADER NAME OVERLAY ###
	vice_leader_name_gui = {
		context_type = player_context
		window_name = "vice_leader_name_container"
		parent_window_token = politics_tab
		
		visible = {
			is_ai = no
			has_country_flag = show_vice_leader_window
			has_vice_leader = yes
		}
	}
}
