RAJ_muslim_discontent_increase = {
	if = {
		limit = { has_variable = RAJ_muslim_discontent }
		set_temp_variable = { RAJ_muslim_discontent_delta_tt = RAJ_muslim_discontent_delta }
		multiply_temp_variable = { RAJ_muslim_discontent_delta_tt = 100 }
		custom_effect_tooltip = RAJ_muslim_discontent_increase_tt
		subtract_from_variable = { RAJ_muslim_discontent = RAJ_muslim_discontent_delta }
	}
}
RAJ_muslim_discontent_decrease = {

		custom_effect_tooltip = RAJ_muslim_discontent_decrease_tt
		add_to_variable = { RAJ_muslim_discontent = RAJ_muslim_discontent_delta }
		clamp_variable = { var = RAJ_muslim_discontent max = 0 }
	}
}

RAJ_decrease_revolt_risk_one = {
	if = {
		limit = {
			check_variable = {
				var = ENG_RAJ_revolt_risk
				value = 1
				compare = greater_than_or_equals
			}
		}
		subtract_from_variable = {
			var = ENG_RAJ_revolt_risk
			value = 1
		}
		custom_effect_tooltip = RAJ_revolt_risk_decrease_by_one
	}
	else = { add_stability = 0.05 }
}
RAJ_increase_revolt_risk_one = {
	add_to_variable = {
		var = ENG_RAJ_revolt_risk
		value = 1
	}
	custom_effect_tooltip = RAJ_revolt_risk_increase_by_one
}

RAJ_calculate_food = {
	set_variable = { RAJ_var_food_production = RAJ_var_amount_of_farms }
	multiply_variable = { RAJ_var_food_production = RAJ_var_food_pro_farm }

	set_variable = { RAJ_food_production_cost = RAJ_var_amount_of_farms }
	divide_variable = { RAJ_food_production_cost = -2 }

	set_variable = { RAJ_var_food_exported = RAJ_var_food_production }
	multiply_variable = { RAJ_var_food_exported = RAJ_var_food_exported_value }

	set_variable = { RAJ_var_food_needed = RAJ_var_food_consumption }
	add_to_variable = { RAJ_var_food_needed = RAJ_var_food_exported }
	set_variable = { RAJ_var_net_food_gain = RAJ_var_food_production }
	subtract_from_variable = { RAJ_var_net_food_gain = RAJ_var_food_needed }
}

RAJ_calculate_population_consumption = {
	set_variable = { RAJ_var_indian_population = ROOT.max_manpower_k }
	every_subject_country = {
		set_variable = { temp_manpower = max_manpower_k }
		RAJ = { add_to_variable = { RAJ_var_indian_population = PREV.temp_manpower } }
	}

	# ~0.5 kg of grain per person per day, scaled into food units.
	set_variable = { RAJ_var_food_consumption = RAJ_var_indian_population }
	divide_variable = { RAJ_var_food_consumption = 4900000 }	# Mil T to 0.5Kg
}

# Umbrella refresh. Call this single effect from any focus / event / decision
# that changes the food economy
#     RAJ_recalculate_food = yes
RAJ_recalculate_food = {
	RAJ_calculate_population_consumption = yes
	RAJ_calculate_food = yes
	RAJ_calculate_money = yes
}

#     RAJ_init_food_economy = yes
RAJ_init_food_economy = {
	set_variable = { RAJ_var_amount_of_farms = 25 }
	set_variable = { RAJ_var_food_pro_farm = 0.0036 }
	set_variable = { RAJ_var_food_exported_value = 0.05 }
	RAJ_recalculate_food = yes
}
RAJ_cost_of_subsidise = {
	if = {
		limit = { has_completed_focus = RAJ_focus_agricultural_subsidies }
		set_temp_variable = { money_to_gain = -2 }
		add_money_with_tooltip_effect = yes
	}
}

RAJ_init_crop_economy = {
	set_variable = { RAJ_var_amount_of_cashcrop_farms = 37 }
	set_variable = { RAJ_var_cash_pro_cashcrop_farm = 0.6 }
	set_variable = { RAJ_var_money_going_to_britain_value = 0.3 }
}

RAJ_calculate_money = {
	set_variable = { RAJ_var_money_gained = RAJ_var_amount_of_cashcrop_farms }
	multiply_variable = { RAJ_var_money_gained = RAJ_var_cash_pro_cashcrop_farm }

	set_variable = { RAJ_var_money_going_to_britain = RAJ_var_money_gained }
	multiply_variable = { RAJ_var_money_going_to_britain = RAJ_var_money_going_to_britain_value }

	set_variable = { RAJ_food_production_cost = RAJ_var_amount_of_farms }
	divide_variable = { RAJ_food_production_cost = -2 }

	set_variable = { RAJ_var_net_money_gain = RAJ_var_money_gained }
	add_to_variable = { RAJ_var_net_money_gain = RAJ_food_production_cost }
	subtract_from_variable = { RAJ_var_net_money_gain = RAJ_var_money_going_to_britain }
}
