RAJ_muslim_discontent_increase = {
	if = {
		limit = { has_variable = RAJ_muslim_discontent }
		set_temp_variable = { RAJ_muslim_discontent_delta_tt = RAJ_muslim_discontent_delta }
		multiply_temp_variable = { RAJ_muslim_discontent_delta_tt = 100 }
		custom_effect_tooltip = RAJ_muslim_discontent_increase_tt
		subtract_from_variable = { RAJ_muslim_discontent = RAJ_muslim_discontent_delta }
	}
}
RAJ_muslim_discontent_decrease = {
	if = {
		limit = { has_variable = RAJ_muslim_discontent }
		set_temp_variable = { RAJ_muslim_discontent_delta_tt = RAJ_muslim_discontent_delta }
		multiply_temp_variable = { RAJ_muslim_discontent_delta_tt = 100 }
		custom_effect_tooltip = RAJ_muslim_discontent_decrease_tt
		add_to_variable = { RAJ_muslim_discontent = RAJ_muslim_discontent_delta }
		clamp_variable = { var = RAJ_muslim_discontent max = 0 }
	}
}

RAJ_decrease_revolt_risk_one = {
	if = {
		limit = {
			check_variable = {
				var = ENG_RAJ_revolt_risk
				value = 1
				compare = greater_than_or_equals
			}
		}
		subtract_from_variable = {
			var = ENG_RAJ_revolt_risk
			value = 1
		}
		custom_effect_tooltip = RAJ_revolt_risk_decrease_by_one
	}
	else = {
		add_stability = 0.05
	}
}
RAJ_increase_revolt_risk_one = {
	add_to_variable = {
		var = ENG_RAJ_revolt_risk
		value = 1
	}
	custom_effect_tooltip = RAJ_revolt_risk_increase_by_one
}
RAJ_calculate_food = {
	set_variable = {
		var = RAJ_var_food_production
		value = RAJ_var_amount_of_farms
	}

	### This Var exists so that the amount gained can get modified
	set_variable = {
		var = RAJ_var_food_pro_farm
		value = 0.0036
	}

	multiply_variable = {
		var = RAJ_var_food_production
		value = RAJ_var_food_pro_farm
	}

	set_variable = {
		var = RAJ_var_food_exported_value
		value = 0.05
	}

	set_variable = {
		var = RAJ_var_food_exported
		value = RAJ_var_food_production
	}

	multiply_variable = {
		var = RAJ_var_food_exported
		value = RAJ_var_food_exported_value
	}

	set_variable = {
		var = RAJ_var_net_gain
		value = RAJ_var_food_production
	}

	subtract_from_variable = {
		var = RAJ_var_net_gain
		value = RAJ_var_food_exported
	}

	subtract_from_variable = {
		var = RAJ_var_net_gain
		value = RAJ_var_food_exported
	}
}
RAJ_calculate_population_consumption = {
	set_variable = {
		var = RAJ_var_indian_population
		value = ROOT.max_manpower
	}

	### Food consumed pro person equaling around short of half a kg of rice and wheats a day to survive
	set_variable = {
		var = RAJ_var_food_consumption
		value = RAJ_var_indian_population
	}
}
