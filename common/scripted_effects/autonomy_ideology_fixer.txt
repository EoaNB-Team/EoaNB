store_current_ideologies = {
    hidden_effect = {

        set_variable = { var_stored_centrism_val = party_popularity_100@centrism }
        set_variable = { var_stored_traditional_conservatism_val = party_popularity_100@traditional_conservatism }
        set_variable = { var_stored_social_egalitarianism_val = party_popularity_100@social_egalitarianism }
        set_variable = { var_stored_ideology_radical_socialism_val = party_popularity_100@ideology_radical_socialism }
        set_variable = { var_stored_ideology_social_liberalism_val = party_popularity_100@ideology_social_liberalism }
        set_variable = { var_stored_autocracy_val = party_popularity_100@autocracy }
        set_variable = { var_stored_radical_democracy_val = party_popularity_100@radical_democracy }
        set_variable = { var_stored_chauvinist_populism_val = party_popularity_100@chauvinist_populism }
        set_variable = { var_stored_anarchism_val = party_popularity_100@anarchism }
        set_variable = { var_stored_fundamentalism_val = party_popularity_100@fundamentalism }

        if = {
            limit = { has_government = centrism }
            set_variable = { var_stored_current_party_ideology_group = 1 }
        }
        else_if = {
            limit = { has_government = traditional_conservatism }
            set_variable = { var_stored_current_party_ideology_group = 2 }
        }
        else_if = {
            limit = { has_government = social_egalitarianism }
            set_variable = { var_stored_current_party_ideology_group = 3 }
        }
        else_if = {
            limit = { has_government = ideology_radical_socialism }
            set_variable = { var_stored_current_party_ideology_group = 4 }
        }
        else_if = {
            limit = { has_government = ideology_social_liberalism }
            set_variable = { var_stored_current_party_ideology_group = 5 }
        }
        else_if = {
            limit = { has_government = autocracy }
            set_variable = { var_stored_current_party_ideology_group = 6 }
        }
        else_if = {
            limit = { has_government = radical_democracy }
            set_variable = { var_stored_current_party_ideology_group = 7 }
        }
        else_if = {
            limit = { has_government = chauvinist_populism }
            set_variable = { var_stored_current_party_ideology_group = 8 }
        }
        else_if = {
            limit = { has_government = anarchism }
            set_variable = { var_stored_current_party_ideology_group = 9 }
        }
        else_if = {
            limit = { has_government = fundamentalism }
            set_variable = { var_stored_current_party_ideology_group = 10 }
        }
    }
}

load_prevous_ideologies = {
    hidden_effect = {

        set_temp_variable = { temp_popularity_sum = var_stored_centrism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_traditional_conservatism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_social_egalitarianism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_ideology_radical_socialism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_ideology_social_liberalism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_autocracy_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_radical_democracy_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_chauvinist_populism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_anarchism_val }
        add_to_temp_variable = { temp_popularity_sum = var_stored_fundamentalism_val }

        set_temp_variable = { temp_popularity_correction = 100 }
        subtract_from_temp_variable = { temp_popularity_correction = temp_popularity_sum }

        if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 1 } }
            set_politics = { ruling_party = centrism }
            add_to_variable = { var_stored_centrism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 2 } }
            set_politics = { ruling_party = traditional_conservatism }
            add_to_variable = { var_stored_traditional_conservatism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 3 } }
            set_politics = { ruling_party = social_egalitarianism }
            add_to_variable = { var_stored_social_egalitarianism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 4 } }
            set_politics = { ruling_party = ideology_radical_socialism }
            add_to_variable = { var_stored_ideology_radical_socialism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 5 } }
            set_politics = { ruling_party = ideology_social_liberalism }
            add_to_variable = { var_stored_ideology_social_liberalism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 6 } }
            set_politics = { ruling_party = autocracy }
            add_to_variable = { var_stored_autocracy_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 7 } }
            set_politics = { ruling_party = radical_democracy }
            add_to_variable = { var_stored_radical_democracy_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 8 } }
            set_politics = { ruling_party = chauvinist_populism }
            add_to_variable = { var_stored_chauvinist_populism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 9 } }
            set_politics = { ruling_party = anarchism }
            add_to_variable = { var_stored_anarchism_val = temp_popularity_correction }
        }
        else_if = {
            limit = { check_variable = { var_stored_current_party_ideology_group = 10 } }
            set_politics = { ruling_party = fundamentalism }
            add_to_variable = { var_stored_fundamentalism_val = temp_popularity_correction }
        }

        set_popularities = {
            centrism                   = var:var_stored_centrism_val
            traditional_conservatism   = var:var_stored_traditional_conservatism_val
            social_egalitarianism      = var:var_stored_social_egalitarianism_val
            ideology_radical_socialism = var:var_stored_ideology_radical_socialism_val
            ideology_social_liberalism = var:var_stored_ideology_social_liberalism_val
            autocracy                  = var:var_stored_autocracy_val
            radical_democracy          = var:var_stored_radical_democracy_val
            chauvinist_populism        = var:var_stored_chauvinist_populism_val
            anarchism                  = var:var_stored_anarchism_val
            fundamentalism             = var:var_stored_fundamentalism_val
        }

        clear_variable = var_stored_centrism_val
        clear_variable = var_stored_traditional_conservatism_val
        clear_variable = var_stored_social_egalitarianism_val
        clear_variable = var_stored_ideology_radical_socialism_val
        clear_variable = var_stored_ideology_social_liberalism_val
        clear_variable = var_stored_autocracy_val
        clear_variable = var_stored_radical_democracy_val
        clear_variable = var_stored_chauvinist_populism_val
        clear_variable = var_stored_anarchism_val
        clear_variable = var_stored_fundamentalism_val
        clear_variable = var_stored_current_party_ideology_group
    }
}
