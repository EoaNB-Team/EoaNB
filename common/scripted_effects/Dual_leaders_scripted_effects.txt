### VICE LEADER MANAGEMENT SCRIPTED EFFECTS ###
# Use these effects to change vice leaders during the game

# Remove current vice leader (call before setting a new one)
remove_vice_leader = {
	# Tokugawa Shogunate
	if = {
		limit = { has_idea = TSH_VL_Komei }
		remove_ideas = TSH_VL_Komei
	}
	# United Kingdom
	if = {
		limit = { has_idea = ENG_VL_Victoria }
		remove_ideas = ENG_VL_Victoria
	}
	# Greece
	if = {
		limit = { has_idea = GRE_VL_Otto }
		remove_ideas = GRE_VL_Otto
	}	
	# USA
	if = {
		limit = { has_idea = USA_VL_Breckinridge }
		remove_ideas = USA_VL_Breckinridge
	}
	# Sardinia
	if = {
		limit = { has_idea = PIE_VL_Victor }
		remove_ideas = PIE_VL_Victor
	}
	# Austria
	if = {
		limit = { has_idea = AUS_VL_Franz_Joseph_I }
		remove_ideas = AUS_VL_Franz_Joseph_I
	}

	# Add more as needed
	
	clear_variable = VL_status_var
}
