eoanb_score_system_calc = {
    clear_array = global.score_array
    clear_array = global.score_array_position
    clear_array = global.score_tag_array

	every_country = {
		limit = { NOT = { tag = UCT tag = AFR } }
        if = {
            limit = { NOT = { has_variable = prestige_score } }
            set_variable = { prestige_score = 0 }
        }

        add_to_variable = {
            industrial_score = {
                value = num_of_military_factories
                add = num_of_civilian_factories
                add = num_of_naval_factories
                add = resource_produced@wood
                multiply = 0.1
                add = resource_produced@coal
                multiply = 0.45
                add = resource_produced@cotton
                multiply = 0.20
                add = resource_produced@fabric
                multiply = 0.30
                add = resource_produced@iron
                multiply = 0.35
                add = resource_produced@steel
                multiply = 0.45
                add = resource_produced@rubber
                multiply = 0.35
                add = resource_produced@oil
                multiply = 0.70
                add = resource_produced@aluminium
                multiply = 0.60
                add = resource_produced@tungsten
                multiply = 0.45
                add = resource_produced@chromium
                multiply = 0.40
                add = resource_produced@machine_parts
                multiply = 0.55
                add = modifier@industrial_capacity_factory
                add = 1
                multiply = 1.5
                add = modifier@production_speed_industrial_complex_factor
                add = 1
                multiply = 1.5
                add = modifier@production_factory_efficiency_gain_factor
                add = 1
                multiply = 1.5
                add = modifier@consumer_goods_expected_value
                add = 1
                multiply = 1.5
                add = modifier@production_factory_start_efficiency_factor
                add = 1
                multiply = 1.5
            }
        }

        if = {
            limit = { has_idea = agrarian }
            multiply_variable = { industrial_score = 0.50 }
        }
        else_if = {
            limit = { has_idea = industrializing }
            multiply_variable = { industrial_score = 0.75 }
        }
        else_if = {
            limit = { has_idea = industrialized }
            multiply_variable = { industrial_score = 0.85 }
        }
        else_if = {
            limit = { has_idea = electrification }
            multiply_variable = { industrial_score = 0.90 }
        }

        add_to_variable = {
            military_score = {
                value = num_divisions
                multiply = 2
                add = has_war_support
                add = 1
                multiply = 1.5
                add = land_doctrine_level
                multiply = 1.5
                add = land_doctrine_level
                multiply = 2
                add = modifier@max_planning
                add = 1
                multiply = 2
                add = num_ships
                multiply = 2
            }
        }

        if = {
            limit = { has_idea = law_army_tribal_neolithic }
            multiply_variable = { military_score = 0.35 }
        }
        else_if = {
            limit = { has_idea = law_army_tribal_basic }
            multiply_variable = { military_score = 0.45 }
        }
        else_if = {
            limit = { has_idea = law_army_tribal_improved }
            multiply_variable = { military_score = 0.55 }
        }
        else_if = {
            limit = { has_idea = law_army_tribal_advanced }
            multiply_variable = { military_score = 0.60 }
        }
        else_if = {
            limit = { has_idea = law_pre_industrial_army_chinese }
            multiply_variable = { military_score = 0.33 }
        }
        else_if = {
            limit = { has_idea = law_pre_industrial_army }
            multiply_variable = { military_score = 0.65 }
        }
        else_if = {
            limit = { has_idea = doctrine_victorian_army_1 }
            multiply_variable = { military_score = 0.75 }
        }
        else_if = {
            limit = { has_idea = doctrine_victorian_army_5 }
            multiply_variable = { military_score = 0.80 }
        }
        else_if = {
            limit = { has_idea = law_trench_infantry }
            multiply_variable = { military_score = 0.85 }
        }
        else_if = {
            limit = { has_idea = law_combined_arms }
            multiply_variable = { military_score = 0.90 }
        }
        else_if = {
            limit = { has_idea = law_modern_army }
            multiply_variable = { military_score = 0.95 }
        }

        if = {
            limit = { is_puppet = yes }
            multiply_variable = { industrial_score = 0.20 }
            multiply_variable = { military_score = 0.20 }
            OVERLORD = {
                add_to_variable = { industrial_score = PREV.industrial_score }
                add_to_variable = { military_score = PREV.military_score }
            }
        }

        clamp_variable = {
            var = prestige_score
            min = 0
            max = 1000000
        }

        set_variable = {
            overall_score = {
                value = industrial_score
                add = military_score
                multiply = 0.5
                add = prestige_score
                add = tech_score
                clamp = {
                    min = 0
                    max = 1000000
                }
            }
        }

        add_to_array = { global.score_array = overall_score }
        add_to_array = { global.score_tag_array = THIS.id }
        set_temp_variable = { length_temp = global.score_array^num }

        set_variable = { industrial_score = 0 }
        set_variable = { military_score = 0 }
        set_variable = { prestige_score = 0 }
        set_variable = { tech_score = 0 }
        set_variable = { overall_score = 0 }
    }

    for_loop_effect = {
        start = 0
        end = length_temp
        add = 1
        value = v
        break = b

        find_highest_in_array = {
            array = global.score_array
            value = score_array_highest_value
            index = score_array_highest_index
        }
        remove_from_array = { array = global.score_array index = score_array_highest_index }
        add_to_temp_array = { array = global.score_array_temp value = score_array_highest_value }

        set_temp_variable = { tag_temp = global.score_tag_array^score_array_highest_index }
        remove_from_array = { array = global.score_tag_array index = score_array_highest_index }
        add_to_temp_array = { array = global.score_tag_array_temp value = tag_temp }

        add_to_array = { global.score_array_position = v }
        add_to_variable = { global.score_array_position^v = 1 }

        add_to_array = { global.score_array = global.score_array_temp^v }
        add_to_array = { global.score_tag_array = global.score_tag_array_temp^v }
    }

	every_country = {
		limit = { NOT = { tag = UCT tag = AFR } }
        for_loop_effect = {
            start = 0
            end = length_temp
            add = 1
            value = v
            break = b
            if = {
                limit = { check_variable = { global.score_tag_array^v = THIS } }
                set_variable = { score_array_position_var = v }
                set_variable = { b = 0 }
            }
        }
        add_to_variable = { score_array_position_var = 1 }
    }
}

eoanb_ledger_system_calc = {
    clear_array = global.population_array 
    clear_array = global.factories_array
    clear_array = global.divisions_array
    clear_array = global.ships_array

    set_temp_variable = { length_temp = global.score_tag_array^num }
    for_loop_effect = {
        start = 0
        end = length_temp
        add = 1
        value = v  
        add_to_array = { global.population_array = 1 }	
        add_to_array = { global.factories_array = 1 }	
        add_to_array = { global.divisions_array = 1 }
        add_to_array = { global.ships_array = 1 }
    }

    every_country = {
        limit = { NOT = { tag = UCT tag = AFR } }
        for_loop_effect = {
            start = 0
            end = length_temp
            add = 1
            value = v  
            break = b 
            if = {
                limit = { check_variable = { global.score_tag_array^v = THIS } }
                set_variable = { score_array_position_var_real = score_array_position_var }
                subtract_from_variable = { score_array_position_var_real = 1 }
				set_variable = {
					global.population_array^v = {
                        value = max_manpower
                        divide = 1000000
                    }
				}

				set_variable = {
					global.population_array_b^v = {
                        value = global.population_array^v
					    multiply = 0.001
                    }
				}
                set_variable = { global.factories_array^v = num_of_factories }
                set_variable = { global.divisions_array^v = num_divisions }
                set_variable = { global.ships_array^v = num_ships }
                set_variable = { b = 0 }
            }		
        }		
    }
}
