RAD_SOC_modifiers_government = { #this block would go in scripted effects
 
	if = {
		limit = { check_variable = { political_legitimacy < 0.41 } }
		subtract_from_variable = { rad_soc_approval_of_government = 0.10 }
		log = "RadSoc(gov): legitimacy 0-40% (-10%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.51 } }
		subtract_from_variable = { rad_soc_approval_of_government = 0.03 }
		log = "RadSoc(gov): legitimacy 41-50% (-3%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.56 } }
		log = "RadSoc(gov): legitimacy 51-55% (no change) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.61 } }
		add_to_variable = { rad_soc_approval_of_government = 0.01 }
		log = "RadSoc(gov): legitimacy 56-60% (+1%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.66 } }
		add_to_variable = { rad_soc_approval_of_government = 0.03 }
		log = "RadSoc(gov): legitimacy 61-65% (+3%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.71 } }
		add_to_variable = { rad_soc_approval_of_government = 0.05 }
		log = "RadSoc(gov): legitimacy 66-70% (+5%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { check_variable = { political_legitimacy < 0.81 } }
		add_to_variable = { rad_soc_approval_of_government = 0.08 }
		log = "RadSoc(gov): legitimacy 71-80% (+6%) [?rad_soc_approval_of_government]"
	}
	else = {
		add_to_variable = { rad_soc_approval_of_government = 0.10 }
		log = "RadSoc(gov): legitimacy 81-100% (+7%) [?rad_soc_approval_of_government]"
	}
 
	if = {
		limit = { OR = { PS_pol_stab_minus_five = yes  PS_pol_stab_minus_four = yes  PS_pol_stab_minus_three = yes } }
		subtract_from_variable = { rad_soc_approval_of_government = 0.10 }
		log = "RadSoc(gov): stability -5..-3 (-10%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { OR = { PS_pol_stab_minus_two = yes  PS_pol_stab_minus_one = yes } }
		subtract_from_variable = { rad_soc_approval_of_government = 0.03 }
		log = "RadSoc(gov): stability -2..-1 (-3%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { PS_pol_stab_zero = yes }
		subtract_from_variable = { rad_soc_approval_of_government = 0.01 }
		log = "RadSoc(gov): stability 0 (-1%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { PS_pol_stab_plus_one = yes }
		add_to_variable = { rad_soc_approval_of_government = 0.02 }
		log = "RadSoc(gov): stability +1 (+2%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { PS_pol_stab_plus_two = yes }
		add_to_variable = { rad_soc_approval_of_government = 0.05 }
		log = "RadSoc(gov): stability +2 (+4%) [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { PS_pol_stab_plus_three = yes }
		add_to_variable = { rad_soc_approval_of_government = 0.10 }
		log = "RadSoc(gov): stability +3 (+6%) [?rad_soc_approval_of_government]"
	}
}

setup_rad_soc_sog = {
	# RadSoc
	set_variable = { rad_soc_approval_of_government = 0 }
	
	if = {
		limit = { is_in_array = { parl_gov_coalitions = 1 } }
		log = "RadSoc is in the GOVERNMENT block"
	
		if = {	# Coalition has only one member
			limit = { check_variable = { parl_gov_coalitions^num = 1 } }
			set_variable = { rad_soc_approval_of_government = 1 }
			log = "Government coalition has one member: RadSoc [?rad_soc_approval_of_government]"
		}
		else_if = {	# RadSoc itself is the largest party
			limit = { check_variable = { var = highest_party_ideology value = token:ideology_radical_socialism compare = equals } }
			set_variable = { rad_soc_approval_of_government = 1 }
			log = "RadSoc has the most support in the coalition [?rad_soc_approval_of_government]"
		}
		else = {	# Base support by ideological proximity to the coalition leader
			if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:social_egalitarianism compare = equals }
					is_in_array = { parl_gov_coalitions = 3 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 65 max = 70 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Leader is Social Egalitarianism (close to RadSoc) [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:ideology_social_liberalism compare = equals }
					is_in_array = { parl_gov_coalitions = 2 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 55 max = 60 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Leader is Social Liberalism [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:radical_democracy compare = equals }
					is_in_array = { parl_gov_coalitions = 6 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 40 max = 50 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Leader is Peripheral/Radical Democracy [?rad_soc_approval_of_government]"
			}
			else_if = {	# A close ideology is present but isn't the largest party
				limit = {
					OR = {
						is_in_array = { parl_gov_coalitions = 2 }
						is_in_array = { parl_gov_coalitions = 3 }
						is_in_array = { parl_gov_coalitions = 6 }
					}
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 30 max = 50 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Close parties are in the coalition but none lead [?rad_soc_approval_of_government]"
			}
			else = {
				set_rad_soc_to_opposition = yes	# VERIFY: original said set_rad_dem_to_opposition???
				log = "RadSoc moved to opposition (no close parties in coalition)"
			}
	
			RAD_SOC_modifiers_government = yes
		}
	
		clamp_variable = { var = rad_soc_approval_of_government min = 0 max = 1 }
		log = "Final GOVERNMENT support for RadSoc [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { is_in_array = { parl_neu_coalitions = 1 } }
		log = "RadSoc is in the NEUTRAL block"
	
		if = {	# Largest party excluding the ruling party
			limit = { check_variable = { var = highest_party_ideology@exclude_ruling_party value = token:ideology_radical_socialism compare = equals } }
			set_variable_to_random = { var = rad_soc_approval_of_government min = 10 max = 20 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
			log = "RadSoc is largest besides the ruling party [?rad_soc_approval_of_government]"
		}
		else = {	# Base support by proximity to the government leader
			if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:social_egalitarianism compare = equals }
					is_in_array = { parl_gov_coalitions = 3 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 25 max = 035 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Gov led by Social Egalitarianism (close to RadSoc) [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:ideology_social_liberalism compare = equals }
					is_in_array = { parl_gov_coalitions = 2 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 20 max = 30 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Gov led by Social Liberalism (close to RadSoc) [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					OR = {
						is_in_array = { parl_gov_coalitions = 2 }
						is_in_array = { parl_gov_coalitions = 3 }
					}
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 15 max = 25 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Close parties are in the government [?rad_soc_approval_of_government]"
			}
			else = {
				set_variable_to_random = { var = rad_soc_approval_of_government min = 15 max = 20 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "No close ideologies in the government [?rad_soc_approval_of_government]"
			}
	
			RAD_SOC_modifiers_opp_neutral = yes
		}
	
		clamp_variable = { var = rad_soc_approval_of_government min = 0 max = 1 }
		log = "Final NEUTRAL support for RadSoc [?rad_soc_approval_of_government]"
	}
	else_if = {
		limit = { is_in_array = { parl_opp_coalitions = 1 } }
		log = "RadSoc is in the OPPOSITION block"
	
		if = {	# Largest party excluding the ruling party
			limit = { check_variable = { var = highest_party_ideology@exclude_ruling_party value = token:ideology_radical_socialism compare = equals } }
			set_variable_to_random = { var = rad_soc_approval_of_government min = 0 max = 15 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
			log = "RadSoc is largest besides the ruling party [?rad_soc_approval_of_government]"
		}
		else = {	# Base support by proximity to the government leader
			if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:social_egalitarianism compare = equals }
					is_in_array = { parl_gov_coalitions = 3 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 25 max = 30 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Gov led by Social Egalitarianism (close to RadSoc) [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					check_variable = { var = highest_party_ideology value = token:ideology_social_liberalism compare = equals }
					is_in_array = { parl_gov_coalitions = 2 }
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 15 max = 25 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Gov led by Social Liberalism (close to RadSoc) [?rad_soc_approval_of_government]"
			}
			else_if = {
				limit = {
					OR = {
						is_in_array = { parl_gov_coalitions = 2 }
						is_in_array = { parl_gov_coalitions = 3 }
					}
				}
				set_variable_to_random = { var = rad_soc_approval_of_government min = 5 max = 15 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "Close parties are in the government [?rad_soc_approval_of_government]"
			}
			else = {
				set_variable_to_random = { var = rad_soc_approval_of_government min = 5 max = 10 integer = yes }
				divide_variable = { rad_soc_approval_of_government = 100 }
				log = "No one close to RadSoc in the government [?rad_soc_approval_of_government]"
			}
	
			RAD_SOC_modifiers_opp_neutral = yes
		}
	
		clamp_variable = { var = rad_soc_approval_of_government min = 0 max = 1 }
		log = "Final OPPOSITION support for RadSoc [?rad_soc_approval_of_government]"
	}
	else = {
		set_variable = { rad_soc_approval_of_government = 0 }
		log = "RadSoc is banned or unorganised (Support: 0%)"
	}
}

setup_soc_lib_sog = {
	# Soclib
	set_variable = { soc_lib_approval_of_government = 0 }
	
	if = {
		limit = { is_in_array = { parl_gov_coalitions = 2 } }
		log = "Soclib is in the GOVERNMENT block"
	
		if = {	# Coalition has only one member
			limit = { check_variable = { parl_gov_coalitions^num = 1 } }
			set_variable = { soc_lib_approval_of_government = 1 }
			log = "Government coalition has one member: Soclib [?soc_lib_approval_of_government]"
		}
		else_if = {	# Soclib itself is the largest party
			limit = { check_variable = { var = highest_party_ideology value = token:ideology_social_liberalism compare = equals } }
			set_variable = { soc_lib_approval_of_government = 1 }
			log = "Soclib has the most support in the coalition [?soc_lib_approval_of_government]"
		}
		else = {	# Base support by ideological proximity to the coalition leader
			if = {
				limit = { 
					check_variable = { var = highest_party_ideology value = token:social_egalitarianism compare = equals } 
					is_in_array = { parl_gov_coalitions = 3 }
				}
				set_variable_to_random = { var = soc_lib_approval_of_government min = 65 max = 70 integer = yes }
				divide_variable = { soc_lib_approval_of_government = 100 }
				log = "Gov led by Social Egalitarianism (close to Soclib) [?soc_lib_approval_of_government]"
			}
			else_if = {
				limit = { 
					check_variable = { var = highest_party_ideology value = token:centrism compare = equals } 
					is_in_array = { parl_gov_coalitions = 4 }
				}
				set_variable_to_random = { var = soc_lib_approval_of_government min = 50 max = 60 integer = yes }
				divide_variable = { soc_lib_approval_of_government = 100 }
				log = "Gov led by Conservative Liberalism (close to Soclib) [?soc_lib_approval_of_government]"
			}
			else_if = {
				limit = { 
					OR = {
						is_in_array = { parl_gov_coalitions = 3 } 
						is_in_array = { parl_gov_coalitions = 4 } 
					}
				}

			}

		}

	set_variable = { chauv_pop_approval_of_government = 0 }
	
	if = {
		limit = { is_in_array = { parl_gov_coalitions = 7 } }
		log = "Chauvpop is in the GOVERNMENT block"
	
		if = {	# Coalition has only one member
			limit = { check_variable = { parl_gov_coalitions^num = 1 } }
			set_variable = { chauv_pop_approval_of_government = 1 }
			log = "Government coalition has one member: Chauvpop [?chauv_pop_approval_of_government]"
		}
		else_if = {	# Chauvpop itself is the largest party
			limit = { check_variable = { var = highest_party_ideology value = token:chauvinist_populism compare = equals } }
			set_variable = { chauv_pop_approval_of_government = 1 }
			log = "Chauvpop has the most support in the coalition [?chauv_pop_approval_of_government]"
		}
		else = {	# Base support by ideological proximity to the coalition leader
			if = {
				limit = { 
					check_variable = { var = highest_party_ideology value = token:autocracy compare = equals } 
					is_in_array = { parl_gov_coalitions = 8 }
				}

				log = "Close parties are in the government [?fun_approval_of_government]"
			}
			else = { 
				set_variable_to_random = { var = fun_approval_of_government min = 25 max = 35 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
				log = "No close ideologies in the government [?fun_approval_of_government]"
			}
	
			FUN_modifiers_opp_neutral = yes
		}
	
		clamp_variable = { var = fun_approval_of_government min = 0 max = 1 }
		log = "Final NEUTRAL support for Fun [?fun_approval_of_government]"
	}
	else_if = {
		limit = { is_in_array = { parl_opp_coalitions = 9 } }
		log = "Fun is in the OPPOSITION block"
	
		if = {	# Largest party excluding the ruling party
			limit = { check_variable = { var = highest_party_ideology@exclude_ruling_party value = token:fundamentalism compare = equals } }
			set_variable_to_random = { var = fun_approval_of_government min = 25 max = 30 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
			log = "Fun is largest besides the ruling party [?fun_approval_of_government]"
		}
		else = {	# Base support by proximity to the government leader
			if = {
				limit = { 
					check_variable = { var = highest_party_ideology value = token:traditional_conservatism compare = equals } 
					is_in_array = { parl_gov_coalitions = 5 }
				}
				set_variable_to_random = { var = fun_approval_of_government min = 35 max = 40 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
				log = "Gov led by Traditional Conservativism (close to Fun) [?fun_approval_of_government]"
			}
			else_if = {
				limit = { 
					check_variable = { var = highest_party_ideology value = token:autocracy compare = equals } 
					is_in_array = { parl_gov_coalitions = 8 }
				}
				set_variable_to_random = { var = fun_approval_of_government min = 30 max = 35 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
				log = "Gov led by Authoritarianism (close to Fun) [?fun_approval_of_government]"
			}
			else_if = {
				limit = { 
					OR = {
						is_in_array = { parl_gov_coalitions = 8 }
						is_in_array = { parl_gov_coalitions = 5 }
					} 
				}

				set_variable_to_random = { var = fun_approval_of_government min = 25 max = 32 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
				log = "Close parties are in the government [?fun_approval_of_government]"
			}
			else = { 
				set_variable_to_random = { var = fun_approval_of_government min = 10 max = 15 integer = yes }
				divide_variable = { fun_approval_of_government = 100 }
				log = "No close ideologies in the government [?fun_approval_of_government]"
			}
	
			FUN_modifiers_opp_neutral = yes
		}
	
		clamp_variable = { var = fun_approval_of_government min = 0 max = 1 }
		log = "Final OPPOSITION support for Fun [?fun_approval_of_government]"
	}
	else = {
		set_variable = { fun_approval_of_government = 0 }
		log = "Fun is banned or unorganised (Support: 0%)"
	}
}

add_rad_soc_sog_with_tooltip_effect = {
	if = {
		limit = { NOT = { check_variable = { var = highest_party_ideology value = token:ideology_radical_socialism compare = equals } } }
		if = {
			limit = { check_variable = { sog_to_gain < 0 } } #Checks if the values are positive
			if = { #To decide if the party is in the government -> decides between loyalty and support
				limit = { is_in_array = { parl_gov_coalitions = 1 } }
				custom_effect_tooltip = LOSE_SOG_RAD_SOC_LOYALTY_TT #Gives a TT for losing sog
			}
			else = { custom_effect_tooltip = LOSE_SOG_RAD_SOC_TT }
		}
		else = {
			if = { #To decide if the party is in the government -> decides between loyalty and support
				limit = { is_in_array = { parl_gov_coalitions = 1 } }
				custom_effect_tooltip = GAIN_SOG_RAD_SOC_LOYALTY_TT #Gives a TT for gaining sog
			}
			else = { custom_effect_tooltip = GAIN_SOG_RAD_SOC_TT }
		}
		hidden_effect = { #This section will give the values specified on the temp variable to the actual variable
			set_temp_variable = { rad_soc_sog_temp = rad_soc_approval_of_government } 
			add_to_variable = { rad_soc_approval_of_government = sog_to_gain }
			clamp_variable = { #Keeps the value of the var between 0 and 1 (0 and 100)
				var = rad_soc_approval_of_government
				min = 0
				max = 1
			}
		}
	}

set_all_opposition_to_neutral = {
	for_each_loop = {
	    array = parl_opp_coalitions
	    value = parl_v
	    index = parl_i
	    
		if = {
			limit = { NOT = { is_in_array = { parl_neu_coalitions = parl_v } } }
			add_to_array = { parl_neu_coalitions = parl_v }
		}
		remove_from_array = {
			array = parl_opp_coalitions
			index = parl_i
		}
	}
}

reverse_coalitions = {
	# Extracts the two arrays into temp_arrays
	for_each_loop = {
	    array = parl_gov_coalitions
	    value = parl_v
	    index = parl_i
	    
	    add_to_temp_array = { parl_gov_temp = parl_v }
		remove_from_array = {
			array = parl_gov_coalitions
			index = parl_i
		}
	}
	for_each_loop = {
	    array = parl_opp_coalitions
	    value = parl_v
	    index = parl_i
	    
	    add_to_temp_array = { parl_opp_temp = parl_v }
		remove_from_array = {
			array = parl_opp_coalitions
			index = parl_i
		}
	}

	# Clears the original arrays
	clear_array = parl_gov_coalitions
	clear_array = parl_opp_coalitions

	# Adds the temp_arrays into the original arrays
	for_each_loop = {
		array = parl_gov_temp
	    value = parl_v
	    index = parl_i
	    
	    add_to_array = { parl_gov_coalitions = parl_v }
	}
	for_each_loop = {
		array = parl_opp_temp
	    value = parl_v
	    index = parl_i
	    
	    add_to_array = { parl_opp_coalitions = parl_v }
	}
}
