##### SOG (Support of Government) Setup #####

# The values are calulated based on government parties, political stability and government legitimacy 
# If it ever goes south, the values are logged in logs/game.log so you can see why the values were calculated the way they were, but you can ping @Syrexx

# The next scripted effects should be executed on coalition changes
setup_rad_soc_sog = {
	# RadSoc
	if = {
		limit = { is_in_array = { parl_gov_coalitions = 1 } }
		set_variable = { rad_soc_approval_of_government = 0 }
		if = { #Checks if the coalition only has one member
			limit = { check_variable = { parl_gov_coalitions^num = 1} }
			set_variable = { rad_soc_approval_of_government = 1 }
			log = "The government coalition only has one member, RadSoc (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
		}
		else = {
			if = { # Checks if party has the highest support
				limit = { check_variable = { var = highest_party_ideology value = token:ideology_radical_socialism compare = equals } }
				set_variable = { rad_soc_approval_of_government = 1 }
				log = "RadSoc has the most support in the government coalition (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
			}
			else = { #ideological difference between the coalition leader
				if = {
					limit = { 
						check_variable = { var = highest_party_ideology value = token:social_egalitarianism compare = equals } 
						is_in_array = { parl_gov_coalitions = 3 }
					}
					set_variable_to_random = {
						var = rad_soc_approval_of_government
						min = 0.65
						max = 0.70
					}
					log = "RadSoc don't have the most support, but Social Egalitarianism has it (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else_if = {
					limit = { 
						check_variable = { var = highest_party_ideology value = token:ideology_social_liberalism compare = equals }
						is_in_array = { parl_gov_coalitions = 2 }
					}
					set_variable_to_random = {
						var = rad_soc_approval_of_government
						min = 0.55
						max = 0.60
					}
					log = "RadSoc don't have the most support, but Social Liberalism has it (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else_if = {
					limit = { 
						check_variable = { var = highest_party_ideology value = token:radical_democracy compare = equals }
						is_in_array = { parl_gov_coalitions = 6 } 
					}
					set_variable_to_random = {
						var = rad_soc_approval_of_government
						min = 0.40
						max = 0.50
					}
					log = "RadSoc don't have the most support, but Peripheral has it (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else_if = {
					limit = {
						OR = { # when ideologies close together are in the coalition but none of them are the biggest party
							is_in_array = { parl_gov_coalitions = 2 }
							is_in_array = { parl_gov_coalitions = 3 }
							is_in_array = { parl_gov_coalitions = 6 }
						}
					}

					set_variable_to_random = {
						var = rad_soc_approval_of_government
						min = 0.30
						max = 0.50
					}
					log = "RadSoc and its similar parties don't have the highest support, but they are together in the coalition (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else = { 
					set_rad_dem_to_opposition = yes 
					log = "RadSoc has been moved to opposition (they don't have similar parties in the coalition.)"
				}

				if = { # The value will be increased/decreased based on the political stability and the legitimacy
					limit = { 
						AND = { 
							check_variable = { var = political_legitimacy value = 0 compare = greater_than_or_equals}
							check_variable = { var = political_legitimacy value = 0.40 compare = less_than_or_equals} 
						}
					}
					subtract_from_variable = { rad_soc_approval_of_government = 0.10 }
					log = "You have legitimacy between 0 and 40% (-10% loyalty) (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else_if = {
					limit = { 
						AND = { 
							check_variable = { var = political_legitimacy value = 0.41 compare = greater_than_or_equals}
							check_variable = { var = political_legitimacy value = 0.50 compare = less_than_or_equals} 
						}
					}
					subtract_from_variable = { rad_soc_approval_of_government = 0.03 }
					log = "You have legitimacy between 41 and 50% (-3% loyalty) (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				# If the legitimacy is between 51 and 55% the value won't change
				else_if = {
					limit = { 
						AND = { 
							check_variable = { var = political_legitimacy value = 0.56 compare = greater_than_or_equals}
							check_variable = { var = political_legitimacy value = 0.60 compare = less_than_or_equals} 
						}
					}
					add_to_variable = { rad_soc_approval_of_government = 0.01 }

					}
					add_to_variable = { rad_soc_approval_of_government = 0.03 }
					log = "You have legitimacy between 61 and 65% (+3% loyalty) (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"
				}
				else_if = {
					limit = { 
						AND = { 
							check_variable = { var = political_legitimacy value = 0.66 compare = greater_than_or_equals}
							check_variable = { var = political_legitimacy value = 0.70 compare = less_than_or_equals}

					}

					log = "You have legitimacy between 66 and 70% (+5% loyalty) (rad_soc_approval_of_government : [?rad_soc_approval_of_government])"

### SOG Value Change
# How to use it:
# set_temp_variable = { sog_to_gain = 0.xx }
# add_<ideology_name>_sog_with_tooltip_effect = yes

add_rad_soc_sog_with_tooltip_effect = {
	if = {
		limit = { NOT = { check_variable = { var = highest_party_ideology value = token:ideology_radical_socialism compare = equals } } }
		if = {
			limit = { check_variable = { sog_to_gain < 0 } } #Checks if the values are positive
			if = { #To decide if the party is in the government -> decides between loyalty and support
				limit = { is_in_array = { parl_gov_coalitions = 1 } }
				custom_effect_tooltip = LOSE_SOG_RAD_SOC_LOYALTY_TT #Gives a TT for losing sog
			}
			else = { custom_effect_tooltip = LOSE_SOG_RAD_SOC_TT }
		}
		else = {
			if = { #To decide if the party is in the government -> decides between loyalty and support
				limit = { is_in_array = { parl_gov_coalitions = 1 } }
				custom_effect_tooltip = GAIN_SOG_RAD_SOC_LOYALTY_TT #Gives a TT for gaining sog
			}
			else = { custom_effect_tooltip = GAIN_SOG_RAD_SOC_TT }
		}
		hidden_effect = { #This section will give the values specified on the temp variable to the actual variable
			set_temp_variable = { rad_soc_sog_temp = rad_soc_approval_of_government } 
			add_to_variable = { rad_soc_approval_of_government = sog_to_gain }
			clamp_variable = { #Keeps the value of the var between 0 and 1 (0 and 100)
				var = rad_soc_approval_of_government
				min = 0
				max = 1
			}
		}
	}
}

	    value = parl_v
	    index = parl_i
	    
		if = {
			limit = { NOT = { is_in_array = { parl_neu_coalitions = parl_v } } }
			add_to_array = { parl_neu_coalitions = parl_v }
		}
		remove_from_array = {
			array = parl_opp_coalitions
			index = parl_i
		}
	}
}

reverse_coalitions = {
	# Extracts the two arrays into temp_arrays
	for_each_loop = {
	    array = parl_gov_coalitions
	    value = parl_v
	    index = parl_i
	    
	    add_to_temp_array = { parl_gov_temp = parl_v }
		remove_from_array = {
			array = parl_gov_coalitions
			index = parl_i
		}
	}
	for_each_loop = {
	    array = parl_opp_coalitions
	    value = parl_v
	    index = parl_i
	    
	    add_to_temp_array = { parl_opp_temp = parl_v }
		remove_from_array = {
			array = parl_opp_coalitions
			index = parl_i
		}
	}

	# Clears the original arrays
	clear_array = parl_gov_coalitions
	clear_array = parl_opp_coalitions

	# Adds the temp_arrays into the original arrays
	for_each_loop = {
		array = parl_gov_temp
	    value = parl_v
	    index = parl_i
	    
	    add_to_array = { parl_gov_coalitions = parl_v }
	}
	for_each_loop = {
		array = parl_opp_temp
	    value = parl_v
	    index = parl_i
	    
	    add_to_array = { parl_opp_coalitions = parl_v }
	}
}
