AFR_colonise_subsume_state = {
	if = {
		limit = { check_variable = { AFR_has_regional_subject = 1 } }
		event_target:AFR_regional_subject = { transfer_state = PREV }
	}
	else = { ROOT = { transfer_state = PREV } }
	add_compliance = 50
}

AFR_colonise_annex_border_state = {
	if = {
		limit = { check_variable = { AFR_has_regional_subject = 1 } }
		event_target:AFR_regional_subject = { transfer_state = PREV }
	}
	else = { ROOT = { transfer_state = PREV } }
	add_resistance = 30
}

AFR_colonise_declare_war_on_capital_owner = {
	owner = {
		if = {
			limit = { check_variable = { AFR_has_regional_subject = 1 } }
			event_target:AFR_regional_subject = {
				declare_war_on = {
					target = PREV
					type = take_claimed_state
				}
				ROOT = {
					add_to_war = {
						targeted_alliance = PREV
						enemy = PREV.PREV
						hostility_reason = asked_to_join
					}
				}
			}
		}
		else = {
			ROOT = {
				declare_war_on = {
					target = PREV
					type = take_claimed_state
				}
			}
		}
	}
}
