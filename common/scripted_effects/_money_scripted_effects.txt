# Internal: adds money_to_gain to money with overflow protection. No tooltip.
# Prereq: set money_to_gain before calling.
_add_money_core = {
	set_temp_variable = { money_temp = money }
	set_variable = {
		money = {
			value = money
			add = money_to_gain
			if = {
				limit = {
					value = {
						value = money_to_gain
						greater_than = 0
					}
					and = {
						value = money
						less_than = money_temp
					}
				}
				pow = 0
				add = 1999999
			}
			else_if = {
				limit = {
					value = {
						value = money_to_gain
						less_than = 0
					}
					and = {
						value = money
						greater_than = money_temp
					}
				}
				pow = 0
				add = -2000001
			}
			clamp = { min = -2000000 max = 2000000 }
		}
	}
}

add_unemployment_effect = {
	clamp_temp_variable = {
		var = unemployment_value
		min = -1
		max = 1
	}
	if = {
		limit = { 
			check_expr = {
				value = current_unemployment
				not_equals = 0
			}
		}
		custom_effect_tooltip = ADD_UNEMPLOYMENT_EFFECT_TT
	}
	hidden_effect = {
		set_variable = {
			current_unemployment = {
				value = current_unemployment
				add = unemployment_value
				clamp = { min = 0 max = 1 }
			}
		}
		update_unemployment_modifier = yes
	}
}

# example usage:
## update_unemployment_modifier = yes
update_unemployment_modifier = {
	set_temp_variable = {
		unemployment_value_modifier_temp = {
			value = 1
			add = unemployment_value_modifier
			add = modifier@unemployment_value_factor
			max = 0
		}
	}
	if = {
		limit = { 
			check_expr = {
				value = current_unemployment
				not_equals = 0
			}
		}
		set_temp_variable = {
			current_unemployment_temp = {
				value = current_unemployment
				multiply = unemployment_value_modifier_temp
			}
		}
		set_variable = {
			unemployment_modifier_negative_var = {
				value = current_unemployment_temp
				multiply = -1
			}
		}
		set_variable = {
			unemployment_modifier_half_negative_var = {
				value = current_unemployment_temp
				divide = -2
			}
		}
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = unemployment_dynamic_modifier } } }
			add_dynamic_modifier = { modifier = unemployment_dynamic_modifier }
		}
		else = { force_update_dynamic_modifier = yes }
	}
	else = {
		clear_variable = current_unemployment
		clear_variable = unemployment_modifier_negative_var
		clear_variable = unemployment_modifier_half_negative_var
		if = {
			limit = { has_dynamic_modifier = { modifier = unemployment_dynamic_modifier } }
			remove_dynamic_modifier = { modifier = unemployment_dynamic_modifier }
		}
	}
}

# example usage:
## set_temp_variable = { modifier_value = 0.1 }

change_unemployment_value_modifier = {

			max = -1

	hidden_effect = {

			unemployment_value_modifier = {
				value = unemployment_value_modifier
				add = modifier_value
				clamp = { min = -20000 max = 20000 }
			}
		}
		update_unemployment_modifier = yes
	}
}

# Internal: core loan logic used by both variants.
# Prereq: set loans_amount. Set _loan_add_inflation = 1 temp var if inflation should be added.
_take_loan_core = {
	add_to_variable = { current_loans_sum = loans_amount }
	set_temp_variable = { money_to_gain = loans_amount }
	_add_money_core = yes
	if = {
		limit = { 
			check_expr = {
				value = _loan_add_inflation
				greater_than = 0
			}
		}
		set_temp_variable = {
			inflation_to_gain = {
				value = 0.001
				multiply = loans_amount
			}
		}
		set_variable = {
			current_inflation = {
				value = current_inflation
				add = inflation_to_gain
				clamp = { min = 0 max = 20000 }
			}
		}
		set_variable = {
			inflation_effect_var = {
				value = current_inflation
				divide = -2
			}
		}
		force_update_dynamic_modifier = yes
	}
	# Overflow check on loans — if loans wrapped negative, go bankrupt
	if = {
		limit = { 
			check_expr = {
				value = current_loans_sum
				less_than = 0
			}
		}
		become_bankrupt_effect = yes
	}
}

# example usage:
## set_temp_variable = { loans_amount = 1 }
## take_loan_effect = yes
take_loan_effect = {
	custom_effect_tooltip = TAKE_LOAN_EFFECT_TT
	hidden_effect = {
		set_temp_variable = { _loan_add_inflation = 1 }
		_take_loan_core = yes
	}
	effect_tooltip = {
		set_temp_variable = { loan_size = 1 }
		add_to_variable = { current_loans_sum = loan_size }
		set_temp_variable = { money_to_gain = loan_size }
		add_money_with_tooltip_effect = yes
		set_temp_variable = { inflation_to_gain = 0.001 }
		add_inflation_with_tooltip_effect = yes
	}
}

## pay_all_loans_effect = yes
pay_all_loans_effect = {
	set_temp_variable = { loan_size = current_loans_sum }
	if = {
		limit = { 
			check_expr = {
				value = money
				less_than = loan_size
			}
		}
		set_temp_variable = {
			loan_size = {
				value = money
				divide = 10
				subtract = 0.5
				round = yes
				multiply = 10
			}
		}
	}
	multiply_temp_variable = { loan_size = -1 }
	add_to_variable = { current_loans_sum = loan_size }
	set_temp_variable = { money_to_gain = loan_size }
	add_money_with_tooltip_effect = yes
}
# example usage:
## become_bankrupt_effect = yes
become_bankrupt_effect = {
	custom_effect_tooltip = REMOVE_ALL_LOANS_GET_RID_OF_DEBTS
	hidden_effect = {
		set_variable = { current_loans_sum = 0 }
		set_variable = { money = 0 }
		# Reset inflation directly to 0
		set_variable = { current_inflation = 0 }
		set_variable = { inflation_effect_var = 0 }
		force_update_dynamic_modifier = yes
	}
	add_political_power = -300
	add_timed_idea = {
		idea = bankruptcy
		days = 180
	}
	hidden_effect = {
		activate_mission = can_become_bankrupt_mission
		remove_mission = can_become_bankrupt_mission
		if = {
			limit = { is_ai = no }
			set_variable = { show_bankruptcy_popup = 1 }
		}
	}
}

	if = {
		limit = { NOT = { has_idea = taxation_level_1 } }
		if = {
			limit = { has_idea = taxation_level_2 }
			swap_ideas = { add_idea = taxation_level_1 remove_idea = taxation_level_2 }
		}
		else_if = {
			limit = { has_idea = taxation_level_3 }
			swap_ideas = { add_idea = taxation_level_2 remove_idea = taxation_level_3 }
		}
		else_if = {
			limit = { has_idea = taxation_level_4 }
			swap_ideas = { add_idea = taxation_level_3 remove_idea = taxation_level_4 }
		}
		else_if = {
			limit = { has_idea = taxation_level_5 }

		}
		set_country_flag = { flag = taxes_changed_recently days = 180 value = 1 }
	}
	else = { custom_effect_tooltip = CANNOT_LOWER_TAXATION_LEVEL_TT }
}
