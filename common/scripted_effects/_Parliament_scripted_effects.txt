	parliament_propose_decision = { 
		custom_effect_tooltip = parliament_propose_decision_tt
		set_variable = { lp_cost@var:parliament_decision_token = parliament_decision_lp_cost }
		set_variable = { money_cost@var:parliament_decision_token = parliament_decision_money_cost }
		set_variable = { icon@var:parliament_decision_token = parliament_decision_icon }
		if = {
			limit = { check_variable = { parliament_decision_category = 1 } }
			add_to_array = {
				array = parliament_decision_array_1
				value = parliament_decision_token
			}
			set_variable = { clicked@var:parliament_decision_token = 0 }
		}
		if = {
			limit = { check_variable = { parliament_decision_category = 2 } } # -> voting decision
			add_to_array = {
				array = parliament_decision_array_2
				value = parliament_decision_token
			}
			set_variable = { clicked@var:parliament_decision_token = 0 }
		}
	}
	parliament_hide_decision = { 
		set_variable = { clicked@var:parliament_decision_token = 0 }
		if = {
			limit = { is_in_array = { parliament_decision_array_1 = parliament_decision_token} }
			remove_from_array = {
				array = parliament_decision_array_1
				value = parliament_decision_token
			}
		}
		if = {
			limit = { is_in_array = { parliament_decision_array_2 = parliament_decision_token} }
			remove_from_array = {
				array = parliament_decision_array_2
				value = parliament_decision_token
			}
		}
	}

	# Starts the voting process for the reform held in the temp variable v
	# (set automatically by the dynamic list of the reform windows in the scripted gui)
	# Pays the reform lp cost, locks the reform, remembers it in current_reform_token and
	# fires the vote setup event. The rest of the voting pipeline lives in
	# _Parliament_voting_scripted_effects.txt and _Parliament_system_events.txt
	parliament_enact_reform = {
		if = {
			limit = {
				NOT = { has_country_flag = parliament_reform_vote_ongoing }
			}

			# pay the legislative power cost of the reform
			if = {
				limit = { check_variable = { lp_cost@var:v > 0 } }
				custom_effect_tooltip = AUS_multinational_empire_enact_decision_lp_cost_tt
				set_temp_variable = { lp_cost = lp_cost@var:v }
				multiply_temp_variable = { lp_cost = -1 }
				add_political_power = var:lp_cost
			}

			# lock the reform in the gui while the vote is ongoing
			set_variable = { clicked@var:v = 1 }

			# remember which reform is being voted on - the whole pipeline reads this
			set_variable = { current_reform_token = v }

			set_country_flag = parliament_reform_vote_ongoing

			custom_effect_tooltip = parliament_enact_reform_tt

			hidden_effect = {
				country_event = { id = parliament_event.10 }
			}
		}
		else = {
			# a vote is already ongoing - refuse the click, do not charge anything
			custom_effect_tooltip = parliament_enact_reform_blocked_tt
			log = "parliament_enact_reform: vote already ongoing, ignored click on [?v.GetTokenLocalizedKey]"
		}
	}
