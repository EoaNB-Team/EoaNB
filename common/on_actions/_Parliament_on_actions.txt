					value = parliament_decision_progress_to_add
				}

					var = parliament_decision_days_left
					min = 0
					max = 100
				}

					limit = {

							}
						}
					}

				}

					var = parliament_vote_days_left

				if = {
					limit = {
						NOT = { check_variable = { parliament_vote_total_days > 0 } }

					}
