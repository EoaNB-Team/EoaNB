	id = nap_france_flavor.32
	title = nap_france_flavor.32.t
	desc = nap_france_flavor.32.d
	
	picture = GFX_FRA_Napoleon_iii_event_picture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		TAG = FRA
		#date > 1861.5.6
		has_country_leader = {
			character = FRA_napoleon_iii
			ruling_only = yes
	  }
	}
	
	option = {
		name = nap_france_flavor.32.a 
		ai_chance = { factor = 100 }
    custom_effect_tooltip = FRA_napoleon_health_activated_tt
    unlock_decision_category_tooltip = FRA_decision_category_health_of_his_majesty
    hidden_effect = {
      set_variable = {
		    var = FRA.napoleon_health
		    value = 30
	    }

	    clamp_variable = {
        var = FRA.napoleon_health
        min = 0
		    max = 100
	    }

      set_variable = {
		    var = FRA.napoleon_unhealthy
		    value = 70
	    }

	    clamp_variable = {
        var = FRA.napoleon_unhealthy
        min = 0
		    max = 100
	    }

      set_country_flag = FRA_napoleon_health_activated
      add_country_leader_trait = FRA_trait_serious_illness
      activate_mission = FRA_mission_napoleon_health_decrease

      set_country_leader_portrait = {
			  portrait = "gfx/leaders/FRA/Portrait_FRA_Napoleon_III_health_3.png"
		  }
    }
	}
}
