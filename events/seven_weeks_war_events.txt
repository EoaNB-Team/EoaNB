			}

				tag = PRS
				OR = {
					has_country_flag = eoanb_flag_avoided_2_schleswig_war

					has_country_flag = eoanb_flag_won_2_schleswig_war_gc
					has_country_flag = eoanb_flag_stopgap_2_schleswig_war
				}
				PRS = { has_war = no }
				date > 1866.1.1
			}
		}
	}

	option = {
		name = sevenweekswar.1.a
		ai_chance = { base = 100 }
		add_political_power = 50

			PRS = { country_event = { days = 150 id = sevenweekswar.2 } }

			FRA = { country_event = { id = nap_france.1234 } }
		}
	}
