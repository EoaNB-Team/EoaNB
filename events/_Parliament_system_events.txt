### PARLIAMENT SYSTEM EVENTS ###

add_namespace = parliament_event
#These events will unlock the Parliament Decisions again

#You are now able to make Small Lobby Effort again
country_event = {
	id = parliament_event.1
	title = parliament_event.1.t
	desc = parliament_event.1.d
	picture = GFX_report_event_constitutional_era_parliament

	is_triggered_only = yes

	option = { #Fantastic
		name = parliament_event.1.a
		ai_chance = { factor = 100 }

		set_temp_variable = { parliament_decision_token = token:parliament_small_lobby_effort }
		set_temp_variable = { parliament_decision_lp_cost = 0 }
		set_temp_variable = { parliament_decision_category = 1 }
		set_temp_variable = { parliament_decision_icon = 5 }
		parliament_propose_decision = yes
	}
}
