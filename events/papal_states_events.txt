# Buy Battleships
country_event = {
	id = event_papacy.102
	title = event_papacy.102.t
	desc = event_papacy.102.d

	fire_only_once = yes
	is_triggered_only = yes			
	
	option = { # Buy from our French Allies
		name = event_papacy.102.a
		ai_chance = { base = 20 }
		complete_national_focus = {focus = PAP_focus_buy_french_gunboats
		use_side_message = yes}
	}
	option = { # British ships are the best
		name = event_papacy.102.b
		ai_chance = { base = 40 }
		complete_national_focus = {focus =PAP_focus_buy_british_steamers
		use_side_message = yes}
	}
	option = { # Prussian ships are the future!
		name = event_papacy.102.c
		ai_chance = { base = 20 }
		complete_national_focus = {focus = PAP_focus_buy_prussian_paddle_steamers
		use_side_message = yes
	}
}
}
