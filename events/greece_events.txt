# Turkish investment event
country_event = {	  
	id = greece.618
	title = greece.618.t
	desc = greece.618.d
	picture = GFX_report_event_gre_lake_kopais

	trigger = {
		tag = GRE
		has_country_flag = GREwe_need_money
	}
	
	mean_time_to_happen = {
		days = 250
		modifier = {
			factor = 1.05
			has_country_flag = GREgreece_is_greek
		}
		modifier = {
			factor = 0.95
			has_country_flag = GREwe_need_money
		}
		modifier = {
			factor = 0.9
			has_active_mission = GRE_mission_land_reform
		}
	}
	
	option = {
		name = greece.618.a  
		ai_chance = { factor = 100 }
		random_list = {
			60 = {#Small Investment
				add_tech_bonus = {
					bonus = 0.05
					category = farming_tech
				}
				add_stability = -0.02
			}
			30 = {
				add_tech_bonus = {
					bonus = 0.1
					category = farming_tech
				}
				add_stability = -0.04
			}
			10 = {
				add_tech_bonus = {
					bonus = 0.3
					uses = 2
					category = farming_tech
				}
				add_stability = -0.07
			}
		}
	}
}
