#event that calculates how much PS should be added or subtracted
country_event = {
	id = pssystem.16
	title = pssystem.16.t
	desc = pssystem.16.d
 
	is_triggered_only = yes
	hidden = yes
 
	immediate = {

		set_temp_variable = { pol_stability_yearly_factor = 0 }
 
		if = {
			limit = { is_historical_focus_on = yes }
			add_to_temp_variable = { pol_stability_yearly_factor = 5 }
		}
		else_if = {
			limit = { is_historical_focus_on = no }
			add_to_temp_variable = { pol_stability_yearly_factor = 3 }
		}
 
		if = {
			limit = { has_border_war = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 1 } }
 
		if = {
			limit = { has_war = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 2 } }
 
		if = {
			limit = { has_civil_war = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 2 } }
 
		if = {
			limit = { is_target_of_coup = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 1 } }
 
		if = {
			limit = { has_stability > 0.8 }
			add_to_temp_variable = { pol_stability_yearly_factor = 2 }
		}
		else_if = {
			limit = { has_stability > 0.6 }
			add_to_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else_if = { limit = { has_stability > 0.4 } }
		else_if = {
			limit = { has_stability > 0.2 }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { subtract_from_temp_variable = { pol_stability_yearly_factor = 2 } }
 
		if = {
			limit = { check_variable = { party_popularity@ruling_party > 0.8 } }
			add_to_temp_variable = { pol_stability_yearly_factor = 2 }
		}
		else_if = {
			limit = { check_variable = { party_popularity@ruling_party > 0.6 } }
			add_to_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else_if = { limit = { check_variable = { party_popularity@ruling_party > 0.4 } } } 
		else_if = {
			limit = { check_variable = { party_popularity@ruling_party > 0.3 } }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else_if = {
			limit = { check_variable = { party_popularity@ruling_party > 0.2 } }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 2 }
		}
		else = { subtract_from_temp_variable = { pol_stability_yearly_factor = 3 } }
 
		if = {
			limit = { is_puppet = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 1 } }
 
		if = {
			limit = { is_subject = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}
		else = { add_to_temp_variable = { pol_stability_yearly_factor = 1 } }
 
		if = {
			limit = { is_in_faction = yes }
			add_to_temp_variable = { pol_stability_yearly_factor = 1 }
		}
 
		if = {
			limit = { PS_accident_factor_over_four = yes }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 3 }
		}
		else_if = {
			limit = { OR = { PS_accident_factor_three = yes  PS_accident_factor_four = yes } }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 2 }
		}
		else_if = {
			limit = { OR = { PS_accident_factor_one = yes  PS_accident_factor_two = yes } }
			subtract_from_temp_variable = { pol_stability_yearly_factor = 1 }
		}

		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > 9 } }
			country_event = { id = pssystem.17 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > 6 } }
			country_event = { id = pssystem.18 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > 3 } }
			country_event = { id = pssystem.19 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > -1 } }
			country_event = { id = pssystem.20 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > -4 } }
			country_event = { id = pssystem.21 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > -7 } }
			country_event = { id = pssystem.22 days = 1 }
		}
		else_if = {
			limit = { check_variable = { pol_stability_yearly_factor > -10 } }
			country_event = { id = pssystem.23 days = 1 }
		}
		else = {
			country_event = { id = pssystem.24 days = 1 }
		}
	}
 
	option = {
		ai_chance = { factor = 100 }
		name = pssystem.16.a
	}
}
