country_event = {
	id = kru.7
	title = kru.7.t
	desc = kru.7.d
	picture = GFX_report_event_soviet_finnish_pact
	
	is_triggered_only = yes
	
	option = {
		name = kru.7.a
		ai_chance = { factor = 100 }
		if = {
			limit = {
				tag = KRU
			}
			KRU = {
				start_border_war = {
					change_state_after_war = no
					attacker = {
						state = 760
						num_provinces = 13
						on_win = kru.8
						on_lose = kru.9
						on_cancel = kru.10
					}
				
					defender = {
						state = 298
						num_provinces = 5
						on_win = kru.12
						on_lose = kru.13
						on_cancel = kru.14
					}
				}
				set_border_war_data = {
					attacker = 760
					defender = 298
					combat_width = 10
				}
			}
		}
		set_country_flag = raiding_neighbour
	}
	
}
