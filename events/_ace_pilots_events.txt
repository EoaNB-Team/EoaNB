add_namespace = ace_event

# New ace has been promoted
country_event = {
	id = ace_event.1
	title = ace_event.1.t
	desc = {
		text = ace_event.1.d
		trigger = {
			NOT = { tag = GER }
			NOT = { tag = ITA }
			NOT = { tag = FRA }
			NOT = { tag = JAP }
			NOT = { tag = USA }
			NOT = { tag = ENG }
			NOT = { tag = SOV }
		}
	}
	desc = {
		text = ace_event.1.d_major
		trigger = {
			OR = {
				tag = GER
				tag = ITA
				tag = FRA
				tag = JAP
				tag = USA
				tag = ENG
				tag = SOV
			}
		}
	}
	picture = GFX_report_event_ace_promoted
	
	is_triggered_only = yes
	
	option = {
		name = ace_event.1.a
		add_war_support = 0.02		
	}
}
