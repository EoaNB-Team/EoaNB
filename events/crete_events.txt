#OTO - The Great Cretan Revolution of 1866 (21.08.1866)
country_event = {
	id = GREcrete.1
	title = GREcrete.1.t
	desc = GREcrete.1.d
	picture = GFX_report_event_cretan_revolution

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = OTO
		182 = { is_owned_and_controlled_by = OTO } #Crete
	}

	immediate = {
		hidden_effect = {
			set_global_flag = cretan_revolution_ongoing_flag
			182 = {
				add_core_of = CET
			}
			1242 = {
				add_core_of = CET
			}
			release = CET
			CET = { 
				add_ideas = CET_idea_cretan_spirit
				set_cosmetic_tag = CET_Cretan_Rebels
				load_oob = CET_rebellion_1
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 500
					producer = GRE
				}
			}
			declare_war_on = {
				target = CET
				type = wargoal_cretan_revolution
			}
			GRE = { country_event = { id = GREcrete.2 days = 1 } } 
		}
	}

	#Declare War on the Rebels
	option = {
		name = GREcrete.1.a
	}

	#Take Control of the Rebels
	option = {
		name = GREcrete.1.b
		trigger = {
			is_ai = no
		}
		CET = {
			change_tag_from = ROOT
		}
	}
}
