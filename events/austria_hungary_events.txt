		} 
	}

	title = austria_hungary.909.t
	desc = austria_hungary.909.d
	picture = GFX_report_event_vienna_1
	
	is_triggered_only = yes 
	
	option = {
		name = austria_hungary.909.a 
		ai_chance = { base = 100 }

		if = {
			limit = {
				has_idea = research_spending_level_1
			}
			swap_ideas = { remove_idea = research_spending_level_1 add_idea = research_spending_level_2 }
