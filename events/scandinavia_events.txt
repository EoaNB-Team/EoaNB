			limit = {
				AND = {
					DEN = {

							has_idea = DEN_idea_lost_second_schleswig_war

						OR = {

							has_idea = DEN_idea_lost_second_schleswig_war_2
							has_idea = DEN_idea_lost_second_schleswig_war_3
