#Invited to the Zollverein by Prussia
country_event = {
	id = prussia.28
	title = prussia.28.t
	desc = prussia.28.d
	picture = GFX_report_event_generic_parliament
	
	is_triggered_only = yes
	
	option = {
		name = prussia.28.a
		ai_chance = { 
			factor = 75
			modifier = {
				factor = 0.5
				tag = PIE 
			}
		}
		add_ideas = PRS_Zollverein_associate
		every_country = {
			limit = {
				OR = {
					has_idea = PRS_Zollverein
					has_idea = PRS_Zollverein_1
					has_idea = PRS_Zollverein_2
				}
			} 
			add_opinion_modifier = { target = ROOT modifier = eoanb_om_zv_foreign_trade }
			reverse_add_opinion_modifier = { target = ROOT modifier = eoanb_om_zv_foreign_trade }
		}
		hidden_effect = {
			PRS = { country_event = { days = 1 id = prussiaforeign.29 } }
			PRS = { add_to_variable = { var = PRS_zollverein_associates value = 1 } }
			PRS = {
				if = {
					limit = {
						AND = {
							check_variable = { var = PRS_zollverein_associates value = 3 compare = equals }
							PRS = { NOT = { has_idea = PRS_Zollverein_foreign_treaties_1 } }
						}
					}
					every_country = {
						limit = {
							OR = {
								has_idea = PRS_Zollverein
								has_idea = PRS_Zollverein_1
								has_idea = PRS_Zollverein_2
							}
						} 
						country_event = { days = 3 id = prussiaforeign.28 }
					}
				}
				else_if = {
					limit = {
						AND = {
							check_variable = { var = PRS_zollverein_associates value = 6 compare = greater_than_or_equals }
							PRS = { has_idea = PRS_Zollverein_foreign_treaties_1 }
						}
					}
					every_country = {
						limit = {
							OR = {
								has_idea = PRS_Zollverein
								has_idea = PRS_Zollverein_1
								has_idea = PRS_Zollverein_2
							}
						} 
						country_event = { days = 3 id = prussiaforeign.28 }
					}
				}
				else_if = {
					limit = {
						PRS = { has_idea = PRS_Zollverein_foreign_treaties_2 }
					}
					clear_variable = PRS_zollverein_associates
				}
			}
		}
	}
	
	option = {
		name = prussia.28.b
		ai_chance = { factor = 25 }
		add_political_power = 25
		hidden_effect = {
			PRS = { country_event = { days = 1 id = prussiaforeign.30 } }
		}
	}
	
}
