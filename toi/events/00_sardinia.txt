					white_peace = LVN
					white_peace = AUS
					news_event = news.1

					set_global_flag = PIE_2IW_Victory

                        add_to_variable = {
                            var = HUN_REVOLT_RISK
                            value = 2  
                        }
                        add_to_variable = {
                            var = SLA_REVOLT_RISK
                            value = 1 
                        }
					}
					release_puppet = PRM
					release_puppet = MOD
					release_puppet = TUS
					TUS = { white_peace = AUS }
					TUS = { white_peace = LVN }
					MOD = { white_peace = AUS }
					MOD = { white_peace = LVN }
					PRM = { white_peace = AUS }
					PRM = { white_peace = LVN }
					# France and puppets peace
					FRA = {
						if = {
							limit = { has_offensive_war_with = AUS }
							white_peace = AUS
						}
						if = {
							limit = { has_offensive_war_with = LVN }
							white_peace = LVN
						}
					}
					ALG = {
						if = {
							limit = { has_offensive_war_with = AUS }
							white_peace = AUS
						}
						if = {
							limit = { has_offensive_war_with = LVN }
							white_peace = LVN
						}
					}
					GBS = {
						if = {
							limit = { has_offensive_war_with = AUS }
							white_peace = AUS
						}
						if = {
							limit = { has_offensive_war_with = LVN }
							white_peace = LVN
						}
					}
