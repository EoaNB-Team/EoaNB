					is_in_home_area = yes

						owner = {
							any_owned_state = {
								free_building_slots = {
									building = industrial_complex
									size > 0
									include_locked = yes
								}
								is_in_home_area = yes
							}
						}
					}
				}
			}

			add_building_construction = {
			type = industrial_complex
			level = 1
			instant_build = yes
			}
		}
	}
	
}
