country_event = {
	id = contain.1
	title = contain.1.t
	desc = contain.1.d
	picture = GFX_report_event_generic_riot
	
	is_triggered_only = yes
	
	option = {
		name = contain.1.a
		clr_country_flag = containment_target
		clr_global_flag = containment_successful
		clr_country_flag = containment_possible_target
		set_country_flag = containment_reperations
		remove_ideas = containment_recent_target
		add_timed_idea = { idea = containment_reperations days = 1825 }
		add_political_power = -150
		add_stability = -0.1
		add_war_support = -0.1
		add_named_threat = {
			threat = -50
			name = contain_capitulated
		}
		every_other_country = {
			limit = {
				has_country_flag = containment_starter
				has_offensive_war = no
			}
			clr_country_flag = containment_starter
		}
		every_other_country = {
			limit = {
				has_country_flag = containment_declined_peace
				has_offensive_war = no
			}
			clr_country_flag = containment_declined_peace
		}
		every_other_country = {
			limit = {
				has_country_flag = left_containment_war
			}
			clr_country_flag = left_containment_war
		}
	}
}

	desc = contain.3.d
	picture = GFX_report_event_generic_riot

	is_triggered_only = yes
	
	option = {
		name = contain.3.a
		clr_country_flag = containment_target_won

		every_other_country = {
			limit = {
				has_country_flag = containment_starter
				has_offensive_war = no
			}
			clr_country_flag = containment_starter
		}
		every_other_country = {
			limit = {
				has_country_flag = containment_declined_peace
				has_offensive_war = no
			}
			clr_country_flag = containment_declined_peace
		}
		every_other_country = {
			limit = {
				has_country_flag = left_containment_war
			}
			clr_country_flag = left_containment_war
		}
		ROOT = {
			if = {
				limit = {
					check_variable = {
						var = aggressive_expansion
						value = 39
						compare = greater_than_or_equals
					}
					check_variable = {
						var = aggressive_expansion
						value = 49
						compare = less_than
					}
				}
				subtract_from_variable = {
					var = aggressive_expansion
					value = 20
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = aggressive_expansion
						value = 49
						compare = greater_than_or_equals
					}
					check_variable = {
						var = aggressive_expansion
						value = 69
						compare = less_than
					}
				}
				subtract_from_variable = {
					var = aggressive_expansion
					value = 40
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = aggressive_expansion
						value = 69
						compare = greater_than_or_equals
					}
					check_variable = {
						var = aggressive_expansion
						value = 89
						compare = less_than
					}
				}
				subtract_from_variable = {
					var = aggressive_expansion
					value = 60
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = aggressive_expansion
						value = 89
						compare = greater_than_or_equals
					}
					check_variable = {
						var = aggressive_expansion
						value = 99
						compare = less_than
					}
				}
				subtract_from_variable = {
					var = aggressive_expansion
					value = 80
				}
			}
			else_if = {
				limit = {
					check_variable = {
						var = aggressive_expansion
						value = 99
						compare = greater_than_or_equals
					}
				}
				subtract_from_variable = {
					var = aggressive_expansion
					value = 90
				}
			}
		}
	}
}
