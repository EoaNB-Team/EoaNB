# yearly on_action by Yard1

on_actions = {
	on_yearly_UCT = {
		#####KDE Scripted Effect firing
		#####First checks to make sure its an even year
		#####Second Checks the Decade (Bi-decadal (?) timeline)
		#####Third Checks the year for bi-yearly firing
		effect = {
			log = "on_yearly_kde"
			set_temp_variable = { tempkdecheck1 = global.year }
			modulo_temp_variable = { var = tempkdecheck1 value = 2 }
			if = {
				limit = { 
					check_variable = { var = tempkdecheck1 value = 0 compare = equals } 
				}
				log = "[GetDateText]:kde_first_check_complete"
				if = {
					limit = { check_variable = { global.year < 1870 } }
					log = "[GetDateText]:kde_second_check_complete"
					if = {
						limit = { check_variable = { global.year < 1860 } }
						UCT = { KDE_bi_yearly_event_fire_1858_to_1859 = yes }
						log = "[GetDateText]:kde_scripted_effect_check_complete"
					}
					else_if = {
						limit = { check_variable = { global.year < 1862 } }
						UCT = { KDE_bi_yearly_event_fire_1860_to_1861 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1864 } }
						UCT = { KDE_bi_yearly_event_fire_1862_to_1863 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1866 } }
						UCT = { KDE_bi_yearly_event_fire_1864_to_1865 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1868 } }
						UCT = { KDE_bi_yearly_event_fire_1866_to_1867 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1870 } }
						UCT = { KDE_bi_yearly_event_fire_1868_to_1869 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1872 } }
						UCT = { KDE_bi_yearly_event_fire_1870_to_1871 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1874 } }
						UCT = { KDE_bi_yearly_event_fire_1872_to_1873 = yes }
					}
					else_if = {
						limit = { check_variable = { global.year < 1876 } }
						
					}
					else_if = {
						limit = { check_variable = { global.year < 1878 } }
						
					}
					else = {
						
					}
				}
				else_if = {
					limit = { check_variable = { global.year < 1890 } }
					if = {
						limit = { check_variable = { global.year < 1872 } }
						
					}
					else_if = {
						limit = { check_variable = { global.year < 1874 } }
						
					}
					else_if = {
						limit = { check_variable = { global.year < 1876 } }
						
					}
					else_if = {
						limit = { check_variable = { global.year < 1878 } }
						
					}
					else = {
						
					}
				}
			}
		}
	}
}
