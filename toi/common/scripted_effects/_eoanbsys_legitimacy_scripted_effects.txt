eoanbsys_legtimacy_reset = {
	randomize_variable = {
		var = political_legitimacy
		min = 0.55
		max = 0.75
	}
	eoanbsys_legitimacy_drift_change = yes
}

	add_to_variable = {
		var = political_legitimacy
		value = mtth:political_legitimacy_drift
		tooltip = political_legitimacy_change_tt
	}
}
