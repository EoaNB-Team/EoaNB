## Effect on 'Looming Debt' Mission
GRE_debt_effect = {
	if = {
		limit = {
			AND = {
				check_variable = {
					var = debt_level
					value = 30
					compare = greater_than_or_equals
				}
				check_variable = {
					var = debt_level
					value = 35
					compare = less_than
				}
			}
		}
		add_stability = -0.025
	}
	else_if = {
		limit = {
			AND = {
				check_variable = {
					var = debt_level
					value = 35
					compare = greater_than_or_equals
				}
				check_variable = {
					var = debt_level
					value = 40
					compare = less_than
				}
			}
		}
		add_stability = -0.05
	}
	else_if = {
		limit = {
			AND = {
				check_variable = {
					var = debt_level
					value = 40
					compare = greater_than_or_equals
				}
				check_variable = {
					var = debt_level
					value = 45
					compare = less_than
				}
			}
		}
		decrease_ps_by_one = yes
		add_stability = -0.05
	}
}

GRE_acquire_train_equipment = {
    if = {
        limit = {
            has_tech = basic_train
            NOT = { has_tech = train_1855 }
        }
        add_equipment_to_stockpile  = {
            type = train_equipment_1
            amount = 25
            producer = GRE
        }
    }
    else_if = {
        limit = {
            has_tech = train_1855
            NOT = { has_tech = train_1865 }
        }
        add_equipment_to_stockpile  = {
            type = train_equipment_2
            amount = 25
            producer = GRE
        }
    }
    else = {
        add_equipment_to_stockpile  = {
            type = train_equipment_3
            amount = 25
            producer = GRE
        }
    }
}

GRE_raise_taxes_effect = {
    if = {
        limit = {
            NOT = {
                OR = {
                    has_government = autocracy
                    mhc_has_military_junta = yes
                }
            }
        }
        decrease_ps_by_one = yes
        if = {
            limit = {
                has_idea = very_low_taxes
            }
            swap_ideas = {
                remove_idea = very_low_taxes
                add_idea = low_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = low_taxes
            }
            swap_ideas = {
                remove_idea = low_taxes
                add_idea = normal_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = normal_taxes
            }
            swap_ideas = {
                remove_idea = normal_taxes
                add_idea = high_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = high_taxes
            }
            swap_ideas = {
                remove_idea = high_taxes
                add_idea = very_high_taxes
            }
            debt_level_decrease_by_2 = yes
        }
    }
    else = {
        if = {
            limit = {
                has_idea = very_low_taxes
            }
            swap_ideas = {
                remove_idea = very_low_taxes
                add_idea = low_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = low_taxes
            }
            swap_ideas = {
                remove_idea = low_taxes
                add_idea = normal_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = normal_taxes
            }
            swap_ideas = {
                remove_idea = normal_taxes
                add_idea = high_taxes
            }
            debt_level_decrease_by_1 = yes
        }
        else_if = {
            limit = {
                has_idea = high_taxes
            }
            swap_ideas = {
                remove_idea = high_taxes
                add_idea = very_high_taxes
            }
            debt_level_decrease_by_2 = yes
        }
    }
}
