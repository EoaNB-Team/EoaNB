contain_aggressive_expansion = {
	enable = {
		any_other_country = {
			check_variable = {
				var = aggressive_expansion
				value = 30
				compare = greater_than_or_equals
			}
		}
	}
	abort = {
		NOT = {
			any_other_country = {
				check_variable = {
					var = aggressive_expansion
					value = 30
					compare = greater_than_or_equals
				}
			}
		}
	}
	
	ai_strategy = {
		type = contain
		id = ROOT
		value = 200
	}

	ai_strategy = {
		type = antagonize
		id = ROOT
		value = 200
	}

	ai_strategy = {
		type = declare_war
		id = ROOT
		value = 250
	}

	ai_strategy = {
		type = alliance
		id = ROOT
		value = -100
	}

	ai_strategy = {
		type = befriend
		id = ROOT
		value = -100
	}
	
}
allies_aggressive_expansion = { #don't get all loveydovey with your allies
	enable = {
		has_war_with = containment_target
		has_war_together_with = ROOT
	}
	abort = {
		NOT = {
			has_war_with = containment_target
			has_war_together_with = ROOT
		}
	}
	
	ai_strategy = {
		type = befriend
		id = ROOT
		value = -50
	}
	
}
denounce_aggressive_expansion = { #Don't be friends with the guy you just called evil
	enable = {
		has_country_flag = containment_denounced_them
		any_other_country = {
			has_country_flag = containment_denounced_us
		}
	}
	abort = {
		NOT = {
			any_other_country = {
				has_country_flag = containment_denounced_us
			}
		}
	}
	
	ai_strategy = {
		type = befriend
		id = ROOT
		value = -50
	}

	ai_strategy = {
		type = alliance
		id = ROOT
		value = -50
	}
	
}
