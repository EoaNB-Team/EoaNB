monthly_manpower = {
	base = 5
	modifier = {
		set_temp_variable = {
			var = monthly_manpower_temp
			value = 0
		}
		all_owned_state = {
			add_to_temp_variable = {
				var = monthly_manpower_temp
				value = mtth:local_monthly_manpower_modifier
			}
		}
	}
	modifier = {
		add = modifier@monthly_manpower_modifier
	}
	modifier = {
		set_temp_variable = {
			var = monthly_manpower_factor_temp
			value = 1
		}
		add_to_temp_variable = {
			var = monthly_manpower_factor_temp
			value = modifier@monthly_manpower_factor
		}
		factor = monthly_manpower_factor_temp
	}
}
local_monthly_manpower = {
	base = 1
	modifier = {
		add = modifier@local_monthly_manpower_modifier
	}
	modifier = {
		set_temp_variable = {
			var = local_monthly_manpower_factor_temp
			value = 1
		}
		add_to_temp_variable = {
			var = local_monthly_manpower_factor_temp
			value = modifier@local_monthly_manpower_factor
		}
		factor = local_monthly_manpower_factor_temp
	}
}
