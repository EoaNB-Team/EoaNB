influence = {
	base = 1
	modifier = {
		add = mtth:national_influence
	}
	modifer = {
		add = mtth:foreign_influence
	}
}
national_influence = {
	base = 100
	modifier = {
		add = modifier@national_influence_modifier
	}
	modifier = {
		set_temp_variable = {
			var = influence_temp
			value = 0
		}
		all_owned_state = {
			add_to_temp_variable = {
				var = influence_temp
				value = modifier@local_influence_modifier
			}
		}
		add = influence_temp
	}
	modifier = {
		factor = modifier@national_influence_factor
	}
}
foreign_influence = {
	base = 1
	modifier = {
		set_temp_variable = {
			var = influence_temp
			value = 0
		}
		for_each_loop = {
			array = foreign_influence_value_array
			value = vv
			add_to_temp_variable = {
				var = influence_temp
				value = vv
			}
		}
		add = influence_temp
	}
	modifier = {
		set_temp_variable = {
			var = influence_temp
			value = 0
		}
		all_owned_state = {
			for_each_loop = {
				array = local_foreign_influence_value_array
				value = vv
				add_to_temp_variable = {
					var = influence_temp
					value = vv
				}
			}
		}
		add = influence_temp
	}
	modifier = {
		add = -1
	}
	modifier = {
		factor = modifier@foreign_influence_factor
	}
}
