						AND = {
							NOT = { tag = TSH }

						}
					}
				}
			}
			else_if = {
				limit = { check_variable = { TSH.TSH_Bakumatsu_Debate_Tensions < 0.01 } }
				if = {
					limit = { tag = TSH }

				}
			}
			else_if = {
				limit = { check_variable = { TSH.TSH_Bakumatsu_Debate_Tensions > 0 } }
				if = {
					limit = { is_subject_of = TSH }
					# Compare from rest of Japan
					set_temp_variable = { TSH_Bakumatsu_Debate_Tensions_Compare = TSH.TSH_Bakumatsu_Debate_Tensions }
					subtract_from_temp_variable = { TSH_Bakumatsu_Debate_Tensions_Compare = TSH_Bakumatsu_Debate_Tensions }
					# Apply weights
					if = {
						limit = { check_variable = { TSH_Bakumatsu_Debate_Tensions_Compare < -0.10 } }
						add_to_variable = { TSH_Bakumatsu_Debate_Tensions = -0.02 }
						add_to_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = -0.01 }
						custom_effect_tooltip = TSH_Bakumatsu_Debate_Tensions_Down2_tt
					}
					else_if = {
						limit = { check_variable = { TSH_Bakumatsu_Debate_Tensions_Compare < 0.00 } }
						add_to_variable = { TSH_Bakumatsu_Debate_Tensions = -0.01 }
						add_to_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = -0.01 }
						custom_effect_tooltip = TSH_Bakumatsu_Debate_Tensions_Down1_tt
					}
					else_if = {
						limit = { check_variable = { TSH_Bakumatsu_Debate_Tensions_Compare > 0.10 } }
						add_to_variable = { TSH_Bakumatsu_Debate_Tensions = 0.02 }
						add_to_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = 0.01 }
						custom_effect_tooltip = TSH_Bakumatsu_Debate_Tensions_Up2_tt
					}
					else_if = {
						limit = { check_variable = { TSH_Bakumatsu_Debate_Tension_Compare > 0.00 } }
						custom_effect_tooltip = TSH_Bakumatsu_Debate_Tensions_Up1_tt
						add_to_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = 0.01 }
						add_to_variable = { TSH_Bakumatsu_Debate_Tensions_Compare = 0.01 }
					}
					TSH_Bakumatsu_Debate_Influence_Country_Modify = yes
					activate_mission = TSH_Mission_Bakumatsu_Modernization_Rising_Tensions
				}
				else_if = {
					limit = { tag = TSH }
					if = {
						limit = { check_variable = { TSH_Bakumatsu_Debate_Tension > 0.99 } }
						# Start the Boshin War
						country_event = shogunate_japan_choshu.1
						clr_global_flag = TSH_Bakumatsu_Debate_Ongoing
						set_global_flag = TSH_Boshin_War_Ongoing
						
					}
					else = {
						# Continue Tensions
						custom_effect_tooltip = TSH_Bakumatsu_Debate_Tensions_Up1_tt
						add_to_variable = { TSH_Bakumatsu_Debate_Tensions_Compare = 0.01 }
						random_owned_controlled_state = {
							limit = { 
								is_in_japan = yes
								check_variable = { TSH_Bakumatsu_Debate_Emperor_Influence < 0.9 }
							}
							set_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = 0.02 }
							TSH_Bakumatsu_Debate_Influence_State_Modify = yes
						}
						random_owned_controlled_state = {
							limit = { 
								is_in_japan = yes
								check_variable = { TSH_Bakumatsu_Debate_Emperor_Influence < 0.9 }
							}
							set_temp_variable = { TSH_Bakumatsu_Debate_Influence_Change = 0.01 }
							TSH_Bakumatsu_Debate_Influence_State_Modify = yes
						}
						random_owned_controlled_state = {
							limit = {

								check_variable = { TSH_Bakumatsu_Debate_Emperor_Influence > 0.1 }
							}

							TSH_Bakumatsu_Debate_Influence_State_Modify = yes
						}
